4//! Direct mapping from `tx_sequence_number → (TransactionDigest, event_count, checkpoint_number)`.
5//!
6//! Row key layout: `[bit_reversed_tx_seq_be_u64]` — 8 bytes. Bit reversal
7//! spreads monotonic transaction sequence numbers across the keyspace while
8//! preserving a reversible point-lookup key.
9//!
10//! `event_count` lets readers enumerate a transaction's event_seqs without
11//! reading the tx row itself — used by unfiltered event listing to discover
12//! exactly the events contributing to a page.
13
14use anyhow::bail;
15use anyhow::{Context, Result};
16use bytes::Bytes;
17use integer_encoding::VarInt;
18use sui_types::digests::TransactionDigest;
19use sui_types::messages_checkpoint::CheckpointSequenceNumber;
20
21pub const NAME: &str = "tx_seq_digest";
22
23pub mod col {
24    /// Raw 32-byte TransactionDigest.
25    pub const DIGEST: &str = "d";
26    /// Big-endian u32 count of events emitted by this transaction.
27    pub const EVENT_COUNT: &str = "e";
28    /// Varint-encoded checkpoint sequence number containing this transaction.
29    pub const CHECKPOINT_NUMBER: &str = "c";
30}
31
32/// Row key: `tx_seq.reverse_bits().to_be_bytes()`.
33pub fn encode_key(tx_seq: u64) -> Vec<u8> {
34    tx_seq.reverse_bits().to_be_bytes().to_vec()
35}
36
37/// Decode a bit-reversed row key.
38pub fn decode_key(key: &[u8]) -> Result<u64> {
39    if key.len() != 8 {
40        anyhow::bail!("tx_seq_digest key not 8 bytes (got {})", key.len());
41    }
42    Ok(u64::from_be_bytes(key.try_into().unwrap()).reverse_bits())
43}
44
45pub fn encode(
46    digest: &TransactionDigest,
47    event_count: u32,
48    checkpoint_number: CheckpointSequenceNumber,
49) -> [(&'static str, Bytes); 3] {
50    [
51        (col::DIGEST, Bytes::from(digest.inner().to_vec())),
52        (
53            col::EVENT_COUNT,
54            Bytes::copy_from_slice(&event_count.to_be_bytes()),
55        ),
56        (
57            col::CHECKPOINT_NUMBER,
58            Bytes::from(checkpoint_number.encode_var_vec()),
59        ),
60    ]
61}
62
63pub fn decode(
64    cells: &[(Bytes, Bytes)],
65) -> Result<(TransactionDigest, u32, CheckpointSequenceNumber)> {
66    let mut digest: Option<TransactionDigest> = None;
67    let mut event_count: u32 = 0;
68    let mut checkpoint_number = None;
69    for (column, value) in cells {
70        if column.as_ref() == col::DIGEST.as_bytes() {
71            let bytes: [u8; 32] = value
72                .as_ref()
73                .try_into()
74                .context("tx_seq_digest digest not 32 bytes")?;
75            digest = Some(TransactionDigest::from(bytes));
76        } else if column.as_ref() == col::EVENT_COUNT.as_bytes() {
77            let bytes: [u8; 4] = value
78                .as_ref()
79                .try_into()
80                .context("tx_seq_digest event_count not 4 bytes")?;
81            event_count = u32::from_be_bytes(bytes);
82        } else if column.as_ref() == col::CHECKPOINT_NUMBER.as_bytes() {
83            checkpoint_number = Some(decode_checkpoint_number_value(value)?);
84        }
85    }
86    Ok((
87        digest.context("tx_seq_digest missing digest column")?,
88        event_count,
89        checkpoint_number.context("tx_seq_digest missing checkpoint_number column")?,
90    ))
91}
92
93pub fn decode_checkpoint_number(cells: &[(Bytes, Bytes)]) -> Result<CheckpointSequenceNumber> {
94    for (column, value) in cells {
95        if column.as_ref() == col::CHECKPOINT_NUMBER.as_bytes() {
96            return decode_checkpoint_number_value(value);
97        }
98    }
99
100    bail!("tx_seq_digest missing checkpoint_number column")
101}
102
103fn decode_checkpoint_number_value(value: &Bytes) -> Result<CheckpointSequenceNumber> {
104    let (checkpoint_number, bytes_read) = u64::decode_var(value.as_ref())
105        .context("tx_seq_digest checkpoint_number is not a valid u64 varint")?;
106    if bytes_read != value.len() {
107        bail!("tx_seq_digest checkpoint_number has trailing bytes");
108    }
109    if checkpoint_number.required_space() != bytes_read {
110        bail!("tx_seq_digest checkpoint_number is not canonical");
111    }
112    Ok(checkpoint_number)
113}

115#[cfg(test)]
116mod tests {
117    use super::*;
118
119    #[test]
120    fn encode_key_layout() {
121        for tx_seq in [0u64, 1, 63, 64, 65, 1_000_000, u64::MAX] {
122            let k = encode_key(tx_seq);
123            assert_eq!(k.len(), 8, "key must be 8 bytes for tx_seq={tx_seq}");
124            assert_eq!(&k, &tx_seq.reverse_bits().to_be_bytes());
125        }
126    }
127
128    #[test]
129    fn encode_decode_round_trip() {
130        for tx_seq in [0u64, 1, 63, 64, 65, 1_000_000, u64::MAX] {
131            let k = encode_key(tx_seq);
132            let got = decode_key(&k).expect("decode must accept freshly encoded key");
133            assert_eq!(got, tx_seq);
134        }
135    }
136
137    #[test]
138    fn encode_decode_row_round_trip() {
139        let digest = TransactionDigest::new([7; 32]);
140        let event_count = 123_456;
141        let checkpoint_number = 300_000_000;
142        let cells = encode(&digest, event_count, checkpoint_number);
143
144        assert_eq!(cells[1].1.as_ref(), &event_count.to_be_bytes());
145        assert_eq!(cells[2].1.len(), 5);
146
147        let cells = cells
148            .into_iter()
149            .map(|(column, value)| (Bytes::from_static(column.as_bytes()), value))
150            .collect::<Vec<_>>();
151        let (decoded_digest, decoded_event_count, decoded_checkpoint_number) =
152            decode(&cells).unwrap();
153        assert_eq!(decoded_digest, digest);
154        assert_eq!(decoded_event_count, event_count);
155        assert_eq!(decoded_checkpoint_number, checkpoint_number);
156        assert_eq!(decode_checkpoint_number(&cells).unwrap(), checkpoint_number);
157    }
158
159    #[test]
160    fn decode_key_rejects_wrong_length() {
161        assert!(decode_key(&[0u8; 7]).is_err());
162        assert!(decode_key(&[0u8; 9]).is_err());
163        assert!(decode_key(&[]).is_err());
164    }
165}
