103        Box::pin(async move {

185    pub async fn new_remote_with_credentials(
186        instance_id: String,
187        project_id: Option<String>,
188        is_read_only: bool,
189        timeout: Option<Duration>,
190        max_decoding_message_size: Option<usize>,
191        client_name: String,
192        registry: Option<&Registry>,
193        app_profile_id: Option<String>,
194        pool_config: PoolConfig,
195        credentials_path: Option<String>,
196    ) -> Result<Self> {
197        let config = pool_config;
198        let policy = if is_read_only {
199            "https://www.googleapis.com/auth/bigtable.data.readonly"
200        } else {
201            "https://www.googleapis.com/auth/bigtable.data"
202        };
203        let token_provider: Arc<dyn TokenProvider> = match credentials_path {
204            Some(path) => Arc::new(gcp_auth::CustomServiceAccount::from_file(&path)?),
205            None => gcp_auth::provider().await?,
206        };
207        let tls_config = ClientTlsConfig::new()
208            .ca_certificate(Certificate::from_pem(include_bytes!("../proto/google.pem")))
209            .domain_name("bigtable.googleapis.com");
210        let mut endpoint = Channel::from_static("https://bigtable.googleapis.com")
211            .http2_keep_alive_interval(Duration::from_secs(30))
212            .keep_alive_timeout(Duration::from_secs(10))
213            .keep_alive_while_idle(true)
214            .tls_config(tls_config)?;
215        endpoint = endpoint.timeout(timeout.unwrap_or(DEFAULT_CHANNEL_TIMEOUT));
216        let project_id = match project_id {
217            Some(p) => p,
218            None => token_provider.project_id().await?.to_string(),
219        };
220        let instance_name = format!("projects/{}/instances/{}", project_id, instance_id);
221        let table_prefix = format!("{}/tables/", instance_name);
222        let primer = BigtablePrimer {
223            instance_name,
224            policy: policy.to_string(),
225            token_provider: Some(token_provider.clone()),
226        };
227        let pool =
228            ChannelPool::new_connected(endpoint, config, Some(Box::new(primer)), registry).await?;
229        let auth_channel = AuthChannel::new(pool, policy.to_string(), Some(token_provider));
230        let client = BigtableInternalClient::new(auth_channel).max_decoding_message_size(
231            max_decoding_message_size.unwrap_or(DEFAULT_MAX_DECODING_MESSAGE_SIZE),
232        );
233        Ok(Self {
234            table_prefix,
235            client,
236            client_name,
237            metrics: registry.map(KvMetrics::new),
238            app_profile_id,
239        })
240    }

242    /// Fetch transactions with an optional column filter for partial reads.
243    /// When `columns` is None, all columns are fetched. When Some, only the
244    /// specified column qualifiers are fetched (e.g. `&["td", "ef", "ts", "cn"]`).
245    pub async fn get_transactions_filtered(
246        &mut self,
247        transactions: &[TransactionDigest],
248        columns: Option<&[&str]>,
249    ) -> Result<Vec<TransactionData>> {
250        let keys = transactions
251            .iter()
252            .map(tables::transactions::encode_key)
253            .collect();
254        let filter = columns.map(|cols| {
255            let pattern = format!("^({})$", cols.join("|"));
256            RowFilter {
257                filter: Some(Filter::ColumnQualifierRegexFilter(pattern.into())),
258            }
259        });
260        let mut result = vec![];
261        for (key, row) in self
262            .multi_get(tables::transactions::NAME, keys, filter)
263            .await?
264        {
265            let digest = TransactionDigest::from(
266                <[u8; 32]>::try_from(key.as_ref())
267                    .context("invalid transaction digest key length")?,
268            );
269            result.push(tables::transactions::decode(digest, &row)?);
270        }
271        Ok(result)
272    }

399    /// CAS write to a watermarks row: write `cells` only when the existing `column` cell is
400    /// absent or has a u64 BE value strictly less than `new_value`. Returns `true` iff the
401    /// write happened (i.e. the new value strictly increased the guarded column). Cells not
402    /// included here retain their existing values in BigTable.
403    pub async fn cas_write_pipeline_watermark_cells(
404        &mut self,
405        pipeline: &str,
406        column: &'static str,
407        new_value: u64,
408        cells: Vec<(&'static str, Bytes)>,
409    ) -> Result<bool> {
410        let mutations = build_set_cell_mutations(cells);
411        let predicate = column_value_at_least_filter(column, u64_be(new_value));
412        // Predicate is "guarded column has a value >= new" → predicate_matched = true means
413        // the existing value blocks the write, so false_mutations are what we want to run.
414        let predicate_matched = self
415            .check_and_mutate_row(
416                tables::watermarks::NAME,
417                tables::watermarks::encode_key(pipeline),
418                Some(predicate),
419                Vec::new(),
420                mutations,
421            )
422            .await?;
423        Ok(!predicate_matched)
424    }

446    /// Returns `true` iff the supplied `chain_id` matches the chain_id stored for `pipeline`.
447    /// On the first call (no chain_id cell yet) writes `chain_id` and returns `true`. The
448    /// chain_id cell is independent of the v1 watermark cells, so this can be invoked before
449    /// `init_watermark`.
450    pub async fn accepts_chain_id(&mut self, pipeline: &str, chain_id: [u8; 32]) -> Result<bool> {
451        use tables::watermarks::col;
452        let mutations =
453            build_set_cell_mutations([(col::CHAIN_ID, Bytes::copy_from_slice(&chain_id))]);
454        let predicate = column_exists_filter(col::CHAIN_ID);
455        // Predicate is "row already has a chain_id" → false_mutations write the new chain_id
456        // when nothing is stored yet.
457        let predicate_matched = self
458            .check_and_mutate_row(
459                tables::watermarks::NAME,
460                tables::watermarks::encode_key(pipeline),
461                Some(predicate),
462                Vec::new(),
463                mutations,
464            )
465            .await?;
466        if !predicate_matched {
467            return Ok(true);
468        }
469        let row = self.get_pipeline_watermark_rows(pipeline).await?;
470        let cell = row
471            .iter()
472            .find_map(|(c, v)| (c.as_ref() == col::CHAIN_ID.as_bytes()).then_some(v))
473            .context("chain_id missing after CAS reported it present")?;
474        let stored: [u8; 32] = cell.as_ref().try_into().map_err(|_| {
475            anyhow::anyhow!(
476                "`{}` column has unexpected length {} (expected 32)",
477                col::CHAIN_ID,
478                cell.len()
479            )
480        })?;
481        Ok(stored == chain_id)
482    }

484    /// Create the row for a pipeline iff no schema-version cell exists yet. Used by
485    /// `init_watermark` for fresh rows and the v0 → v1 bootstrap. Returns `true` iff the
486    /// write happened.
487    pub async fn create_pipeline_watermark_if_absent(
488        &mut self,
489        pipeline: &str,
490        new: &WatermarkV1,
491    ) -> Result<bool> {
492        use tables::watermarks::col;
493        let mut cells = vec![
494            (col::SCHEMA_VERSION, u64_be(tables::watermarks::SCHEMA_V1)),
495            (col::EPOCH_HI, u64_be(new.epoch_hi_inclusive)),
496            (col::TX_HI, u64_be(new.tx_hi)),
497            (col::TIMESTAMP_MS_HI, u64_be(new.timestamp_ms_hi_inclusive)),
498            (col::READER_LO, u64_be(new.reader_lo)),
499            (col::PRUNER_HI, u64_be(new.pruner_hi)),
500            (col::PRUNER_TIMESTAMP_MS, u64_be(new.pruner_timestamp_ms)),
501        ];
502        if let Some(checkpoint) = new.checkpoint_hi_inclusive {
503            cells.push((col::CHECKPOINT_HI, u64_be(checkpoint)));
504            let v0 = WatermarkV0 {
505                epoch_hi_inclusive: new.epoch_hi_inclusive,
506                checkpoint_hi_inclusive: checkpoint,
507                tx_hi: new.tx_hi,
508                timestamp_ms_hi_inclusive: new.timestamp_ms_hi_inclusive,
509            };
510            cells.push((col::WATERMARK_V0, Bytes::from(bcs::to_bytes(&v0)?)));
511        }
512        if let Some(bucket_start_cp) = new.bucket_start_cp {
513            cells.push((col::BUCKET_START_CP, u64_be(bucket_start_cp)));
514        }
515        let mutations = build_set_cell_mutations(cells);
516        let predicate = column_exists_filter(tables::watermarks::col::SCHEMA_VERSION);
517        // Predicate is "row has any schema-version cell" → false_mutations write the new row.
518        let predicate_matched = self
519            .check_and_mutate_row(
520                tables::watermarks::NAME,
521                tables::watermarks::encode_key(pipeline),
522                Some(predicate),
523                Vec::new(),
524                mutations,
525            )
526            .await?;
527        Ok(!predicate_matched)
528    }

578    /// Write pre-built entries to BigTable.
579    ///
580    /// On partial failure (some entries succeed, some fail), returns a `PartialWriteError`
581    /// containing the keys that failed. Callers can retain only the failed keys in their
582    /// batch before retrying.
583    pub async fn write_entries(
584        &mut self,
585        table: &str,
586        entries: impl IntoIterator<Item = Entry>,
587    ) -> Result<()> {
588        let entries: Vec<Entry> = entries.into_iter().collect();
589        if entries.is_empty() {
590            return Ok(());
591        }
592
593        let row_keys: Vec<Bytes> = entries.iter().map(|e| e.row_key.clone()).collect();
594
595        let mut request = MutateRowsRequest {
596            table_name: format!("{}{}", self.table_prefix, table),
597            entries,
598            ..MutateRowsRequest::default()
599        };
600        if let Some(ref app_profile_id) = self.app_profile_id {
601            request.app_profile_id = app_profile_id.clone();
602        }
603        let mut response = self.client.clone().mutate_rows(request).await?.into_inner();
604        let mut failed_keys: Vec<MutationError> = Vec::new();
605
606        while let Some(part) = response.message().await? {
607            for entry in part.entries {
608                if let Some(status) = entry.status
609                    && status.code != 0
610                    && let Some(key) = row_keys.get(entry.index as usize)
611                {
612                    failed_keys.push(MutationError {
613                        key: key.clone(),
614                        code: status.code,
615                        message: status.message,
616                    });
617                }
618            }
619        }
620

648    /// Streaming variant of `read_rows`. Returns rows as they arrive from the
649    /// underlying gRPC stream rather than collecting into a Vec.
650    pub async fn read_rows_stream(
651        &mut self,
652        mut request: ReadRowsRequest,
653        table_name: &str,
654    ) -> Result<impl futures::Stream<Item = Result<(Bytes, Vec<(Bytes, Bytes)>)>> + use<>> {
655        if let Some(ref app_profile_id) = self.app_profile_id {
656            request.app_profile_id = app_profile_id.clone();
657        }
658        let response = self.client.clone().read_rows(request).await?.into_inner();
659        let metrics = self.metrics.clone();
660        let client_name = self.client_name.clone();
661        let table_name = table_name.to_owned();
662
663        Ok(async_stream::try_stream! {
664            // Zero-copy accumulator for cell values. BigTable streams cell data
665            // in chunks, and prost deserializes each chunk.value as a Bytes view
666            // into the gRPC buffer (no allocation).
667            //
668            // - Single chunk (common): stays as Bytes, no copies at all
669            // - Multiple chunks (only for values >1MB): copies into Vec<u8>
670            enum CellValue {
671                Empty,
672                Single(Bytes),
673                Multi(Vec<u8>),
674            }
675
676            impl CellValue {
677                fn extend(&mut self, data: Bytes) {
678                    let prev = std::mem::replace(self, CellValue::Empty);
679                    *self = match prev {
680                        CellValue::Empty => CellValue::Single(data),
681                        CellValue::Single(existing) => {
682                            let mut vec = existing.to_vec();
683                            vec.extend_from_slice(&data);
684                            CellValue::Multi(vec)
685                        }
686                        CellValue::Multi(mut vec) => {
687                            vec.extend_from_slice(&data);
688                            CellValue::Multi(vec)
689                        }
690                    };
691                }
692
693                fn replace(&mut self, data: Bytes) {
694                    *self = CellValue::Single(data);
695                }
696
697                fn into_bytes(self) -> Bytes {
698                    match self {
699                        CellValue::Empty => Bytes::new(),
700                        CellValue::Single(b) => b,
701                        CellValue::Multi(v) => Bytes::from(v),
702                    }
703                }
704            }
705
706            let mut response = response;
707            let mut row_key: Option<Bytes> = None;
708            let mut row = vec![];
709            let mut cell_value = CellValue::Empty;
710            let mut cell_name: Option<Bytes> = None;
711            let mut timestamp = 0i64;
712
713            while let Some(message) = response.message().await? {
714                if let Some(ref metrics) = metrics {
715                    report_bt_stats_inner(metrics, &client_name, &table_name, &message.request_stats);
716                }
717                for chunk in message.chunks.into_iter() {
718                    if !chunk.row_key.is_empty() {
719                        row_key = Some(chunk.row_key);
720                    }
721                    match chunk.qualifier {
722                        Some(qualifier) => {
723                            if let Some(name) = cell_name.take() {
724                                row.push((name, cell_value.into_bytes()));
725                                cell_value = CellValue::Empty;
726                            }
727                            cell_name = Some(Bytes::from(qualifier));
728                            timestamp = chunk.timestamp_micros;
729                            cell_value.extend(chunk.value);
730                        }
731                        None => {
732                            if chunk.timestamp_micros == 0 {
733                                cell_value.extend(chunk.value);
734                            } else if chunk.timestamp_micros >= timestamp {
735                                timestamp = chunk.timestamp_micros;
736                                cell_value.replace(chunk.value);
737                            }
738                        }
739                    }
740                    if chunk.row_status.is_some() {
741                        if let Some(RowStatus::CommitRow(_)) = chunk.row_status {
742                            if let Some(name) = cell_name.take() {
743                                row.push((name, cell_value.into_bytes()));
744                            }
745                            if let Some(key) = row_key.take() {
746                                yield (key, std::mem::take(&mut row));
747                            }
748                        }
749                        row_key = None;
750                        row = vec![];
751                        cell_value = CellValue::Empty;
752                        cell_name = None;
753                    }
754                }
755            }
756        })
757    }

759    pub async fn multi_get(
760        &mut self,
761        table_name: &str,
762        keys: Vec<Vec<u8>>,
763        filter: Option<RowFilter>,
764    ) -> Result<Vec<(Bytes, Vec<(Bytes, Bytes)>)>> {
765        let start_time = Instant::now();
766        let num_keys_requested = keys.len();
767        let result = self.multi_get_internal(table_name, keys, filter).await;
768        let elapsed_ms = start_time.elapsed().as_millis() as f64;
769
770        let Some(metrics) = &self.metrics else {
771            return result;
772        };
773
774        let labels = [&self.client_name, table_name];
775        let Ok(rows) = &result else {
776            metrics.kv_get_errors.with_label_values(&labels).inc();
777            return result;
778        };
779
780        metrics
781            .kv_get_batch_size
782            .with_label_values(&labels)
783            .observe(num_keys_requested as f64);
784
785        if num_keys_requested > rows.len() {
786            metrics
787                .kv_get_not_found
788                .with_label_values(&labels)
789                .inc_by((num_keys_requested - rows.len()) as u64);
790        }
791
792        metrics
793            .kv_get_success
794            .with_label_values(&labels)
795            .inc_by(rows.len() as u64);
796
797        metrics
798            .kv_get_latency_ms
799            .with_label_values(&labels)
800            .observe(elapsed_ms);
801
802        if num_keys_requested > 0 {
803            metrics
804                .kv_get_latency_ms_per_key
805                .with_label_values(&labels)
806                .observe(elapsed_ms / num_keys_requested as f64);
807        }
808
809        result
810    }

1000    /// via a single `multi_get` on the `tx_seq_digest` table.
1001    ///
1002    /// Returns a vector parallel to the input: `None` for any tx_seq that has
1003    /// no row (e.g. not yet indexed).
1004    pub async fn resolve_tx_digests(
1005        &mut self,
1006        tx_sequence_numbers: &[u64],
1007    ) -> Result<Vec<Option<TxSeqDigestData>>> {
1008        if tx_sequence_numbers.is_empty() {
1009            return Ok(Vec::new());
1010        }
1011
1012        let keys: Vec<Vec<u8>> = tx_sequence_numbers
1013            .iter()
1014            .map(|s| tx_seq_digest::encode_key(*s))
1015            .collect();
1016
1017        let rows = self.multi_get(tx_seq_digest::NAME, keys, None).await?;
1018
1019        let mut by_seq: HashMap<u64, TxSeqDigestData> = HashMap::with_capacity(rows.len());

1021            let tx_seq = tx_seq_digest::decode_key(row_key.as_ref())?;

1026                    tx_sequence_number: tx_seq,
1027                    digest,
1028                    event_count,
1029                    checkpoint_number,
1030                },
1031            );
1032        }
1033
1034        Ok(tx_sequence_numbers
1035            .iter()
1036            .map(|s| by_seq.get(s).copied())
1037            .collect())
1038    }

1158    ///
1159    /// Returns `[start_tx, end_tx)` where `start_tx` is the first tx in
1160    /// `start_checkpoint` and `end_tx` is one past the last tx in `end_checkpoint`.
1161    /// Reads checkpoint summaries from the checkpoints table.
1162    pub async fn checkpoint_to_tx_range(
1163        &mut self,
1164        checkpoint_range: Range<u64>,
1165    ) -> Result<Range<u64>> {
1166        use crate::tables::checkpoints;
1167
1168        let start_checkpoint = checkpoint_range.start;
1169        let end_checkpoint = checkpoint_range.end.saturating_sub(1);
1170
1171        let start_tx = if start_checkpoint == 0 {
1172            0u64
1173        } else {
1174            let prev = self
1175                .get_checkpoints_filtered(
1176                    &[start_checkpoint - 1],
1177                    Some(&[checkpoints::col::SUMMARY]),
1178                )
1179                .await?;
1180            let summary = prev
1181                .first()
1182                .and_then(|cp| cp.summary.as_ref())
1183                .context("checkpoint summary not found for start bound")?;
1184            summary.network_total_transactions
1185        };
1186
1187        let end_cps = self
1188            .get_checkpoints_filtered(&[end_checkpoint], Some(&[checkpoints::col::SUMMARY]))
1189            .await?;
1190        let end_summary = end_cps
1191            .first()
1192            .and_then(|cp| cp.summary.as_ref())
1193            .context("checkpoint summary not found for end bound")?;
1194        let end_tx = end_summary.network_total_transactions;

1199    /// Resolve `tx_sequence_number`s to fully-populated transaction rows in
1200    /// two multi_gets: one against `tx_seq_digest` for the digest, one
1201    /// against `transactions` for the row bodies. Tx_seqs that don't resolve
1202    /// (not yet indexed) or whose row is missing are silently dropped.
1203    ///
1204    /// Output ordering is not tx_seq order — callers that need a sorted page
1205    /// should sort by `tx_seq` at the end. Callers should pass a bounded
1206    /// slice (typically `page_size + 1`) so the multi_gets stay well-sized.
1207    pub async fn get_transactions_for_seqs(
1208        &mut self,
1209        seqs: Vec<u64>,
1210        columns: Option<&[&str]>,
1211    ) -> Result<Vec<(u64, crate::TransactionData)>> {
1212        if seqs.is_empty() {
1213            return Ok(Vec::new());
1214        }
1215        let resolved = self.resolve_tx_digests(&seqs).await?;
1216        let mut pairs: Vec<TxSeqDigestData> = Vec::with_capacity(resolved.len());
1217        let mut digests: Vec<TransactionDigest> = Vec::with_capacity(resolved.len());
1218        for row in resolved.into_iter().flatten() {
1219            pairs.push(row);
1220            digests.push(row.digest);
1221        }
1222        if digests.is_empty() {
1223            return Ok(Vec::new());
1224        }
1225        let tx_data_vec = self.get_transactions_filtered(&digests, columns).await?;
1226        let mut by_digest: HashMap<TransactionDigest, crate::TransactionData> =
1227            tx_data_vec.into_iter().map(|tx| (tx.digest, tx)).collect();
1228        Ok(pairs
1229            .into_iter()
1230            .filter_map(|row| {
1231                by_digest
1232                    .remove(&row.digest)
1233                    .map(|tx| (row.tx_sequence_number, tx))
1234            })
1235            .collect())
1236    }

1240    metrics: &KvMetrics,
1241    client_name: &str,
1242    table_name: &str,
1243    request_stats: &Option<RequestStats>,
1244) {
1245    let labels = [client_name, table_name];
1246    if let Some(StatsView::FullReadStatsView(view)) =
1247        request_stats.as_ref().and_then(|r| r.stats_view.as_ref())
1248    {
1249        if let Some(latency) = view
1250            .request_latency_stats
1251            .as_ref()
1252            .and_then(|s| s.frontend_server_latency)
1253        {
1254            if latency.seconds < 0 || latency.nanos < 0 {
1255                return;
1256            }
1257            let duration = Duration::new(latency.seconds as u64, latency.nanos as u32);
1258            metrics
1259                .kv_bt_chunk_latency_ms
1260                .with_label_values(&labels)
1261                .observe(duration.as_millis() as f64);
1262        }
1263        if let Some(iteration_stats) = &view.read_iteration_stats {
1264            metrics
1265                .kv_bt_chunk_rows_returned_count
1266                .with_label_values(&labels)
1267                .inc_by(iteration_stats.rows_returned_count as u64);
1268            metrics
1269                .kv_bt_chunk_rows_seen_count
1270                .with_label_values(&labels)
1271                .inc_by(iteration_stats.rows_seen_count as u64);
1272        }
1273    }
1274}

1325        &mut self,
1326        pipelines: &[&str],
1327    ) -> Result<Option<WatermarkV1>> {
1328        if pipelines.is_empty() {
1329            bail!("at least one watermark pipeline must be provided");
1330        }
1331
1332        let keys: Vec<Vec<u8>> = pipelines
1333            .iter()
1334            .map(|name| tables::watermarks::encode_key(name))
1335            .collect();
1336
1337        let rows = self.multi_get(tables::watermarks::NAME, keys, None).await?;
1338
1339        if rows.len() != pipelines.len() {
1340            return Ok(None);
1341        }
1342
1343        // A row is hidden if `checkpoint_hi_inclusive == None` or
1344        // `checkpoint_hi_inclusive < reader_lo`. If any pipeline is hidden, every consumer
1345        // of this method (RPC/graphql) treats the whole result as missing, resulting in `Ok(None)`.
1346        let mut min_wm: Option<(u64, WatermarkV1)> = None;
1347        for (_, row) in &rows {
1348            let Some(wm) = tables::watermarks::decode_v1(row)? else {
1349                return Ok(None);
1350            };
1351            let Some(cp) = wm.checkpoint_hi_inclusive.filter(|cp| *cp >= wm.reader_lo) else {
1352                return Ok(None);
1353            };
1354            if min_wm.as_ref().is_none_or(|(prev, _)| cp < *prev) {
1355                min_wm = Some((cp, wm));
1356            }
1357        }
1358
1359        Ok(min_wm.map(|(_, wm)| wm))
1360    }
