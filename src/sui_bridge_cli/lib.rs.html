215pub fn make_action(chain_id: BridgeChainId, cmd: &GovernanceClientCommands) -> BridgeAction {
216    match cmd {
217        GovernanceClientCommands::EmergencyButton { nonce, action_type } => {
218            BridgeAction::EmergencyAction(EmergencyAction {
219                nonce: *nonce,
220                chain_id,
221                action_type: *action_type,
222            })
223        }
224        GovernanceClientCommands::UpdateCommitteeBlocklist {
225            nonce,
226            blocklist_type,
227            pubkeys_hex,
228        } => BridgeAction::BlocklistCommitteeAction(BlocklistCommitteeAction {
229            nonce: *nonce,
230            chain_id,
231            blocklist_type: *blocklist_type,
232            members_to_update: pubkeys_hex.clone(),
233        }),
234        GovernanceClientCommands::UpdateLimit {
235            nonce,
236            sending_chain,
237            new_usd_limit,
238        } => {
239            let sending_chain_id =
240                BridgeChainId::try_from(*sending_chain).expect("Invalid sending chain id");
241            BridgeAction::LimitUpdateAction(LimitUpdateAction {
242                nonce: *nonce,
243                chain_id,
244                sending_chain_id,
245                new_usd_limit: *new_usd_limit,
246            })
247        }
248        GovernanceClientCommands::UpdateAssetPrice {
249            nonce,
250            token_id,
251            new_usd_price,
252        } => BridgeAction::AssetPriceUpdateAction(AssetPriceUpdateAction {
253            nonce: *nonce,
254            chain_id,
255            token_id: *token_id,
256            new_usd_price: *new_usd_price,
257        }),
258        GovernanceClientCommands::AddTokensOnSui {
259            nonce,
260            token_ids,
261            token_type_names,
262            token_prices,
263        } => {
264            assert_eq!(token_ids.len(), token_type_names.len());
265            assert_eq!(token_ids.len(), token_prices.len());
266            BridgeAction::AddTokensOnSuiAction(AddTokensOnSuiAction {
267                nonce: *nonce,
268                chain_id,
269                native: false, // only foreign tokens are supported now
270                token_ids: token_ids.clone(),
271                token_type_names: token_type_names.clone(),
272                token_prices: token_prices.clone(),
273            })
274        }
275        GovernanceClientCommands::AddTokensOnEvm {
276            nonce,
277            token_ids,
278            token_addresses,
279            token_prices,
280            token_sui_decimals,
281        } => {
282            assert_eq!(token_ids.len(), token_addresses.len());
283            assert_eq!(token_ids.len(), token_prices.len());
284            assert_eq!(token_ids.len(), token_sui_decimals.len());
285            BridgeAction::AddTokensOnEvmAction(AddTokensOnEvmAction {
286                nonce: *nonce,
287                native: true, // only eth native tokens are supported now
288                chain_id,
289                token_ids: token_ids.clone(),
290                token_addresses: token_addresses.clone(),
291                token_prices: token_prices.clone(),
292                token_sui_decimals: token_sui_decimals.clone(),
293            })
294        }
295        GovernanceClientCommands::UpgradeEVMContract {
296            nonce,
297            proxy_address,
298            implementation_address,
299            function_selector,
300            params,
301        } => {
302            let call_data = match function_selector {
303                Some(function_selector) => encode_call_data(function_selector, params),
304                None => vec![],
305            };
306            BridgeAction::EvmContractUpgradeAction(EvmContractUpgradeAction {
307                nonce: *nonce,
308                chain_id,
309                proxy_address: *proxy_address,
310                new_impl_address: *implementation_address,
311                call_data,
312            })
313        }
314    }
315}

317fn encode_call_data(function_selector: &str, params: &[String]) -> Vec<u8> {
318    let left = function_selector
319        .find('(')
320        .expect("Invalid function selector, no left parentheses");
321    let right = function_selector
322        .find(')')
323        .expect("Invalid function selector, no right parentheses");
324    let param_types = function_selector[left + 1..right]
325        .split(',')
326        .map(|x| x.trim())
327        .collect::<Vec<&str>>();
328
329    assert_eq!(param_types.len(), params.len(), "Invalid number of params");
330
331    let mut tokens = vec![];
332    for (param, param_type) in params.iter().zip(param_types.iter()) {
333        let token = match param_type.to_lowercase().as_str() {
334            "uint256" => {
335                DynSolValue::Uint(U256::from_str_radix(param, 10).expect("Invalid U256"), 256)
336            }
337            "bool" => DynSolValue::Bool(match param.as_str() {
338                "true" => true,
339                "false" => false,
340                _ => panic!("Invalid bool in params"),
341            }),
342            "string" => DynSolValue::String(param.clone()),
343            // TODO: need to support more types if needed
344            _ => panic!("Invalid param type"),
345        };
346        tokens.push(token);
347    }
348
349    let mut call_data = Keccak256::digest(function_selector).digest[0..4].to_vec();
350    if !tokens.is_empty() {
351        call_data.extend(DynSolValue::Tuple(tokens).abi_encode());
352    }
353    call_data
354}

356pub fn select_contract_address(
357    config: &LoadedBridgeCliConfig,
358    cmd: &GovernanceClientCommands,
359) -> EthAddress {
360    match cmd {
361        GovernanceClientCommands::EmergencyButton { .. } => config.eth_bridge_proxy_address,
362        GovernanceClientCommands::UpdateCommitteeBlocklist { .. } => {
363            config.eth_bridge_committee_proxy_address
364        }
365        GovernanceClientCommands::UpdateLimit { .. } => config.eth_bridge_limiter_proxy_address,
366        GovernanceClientCommands::UpdateAssetPrice { .. } => config.eth_bridge_config_proxy_address,
367        GovernanceClientCommands::UpgradeEVMContract { proxy_address, .. } => *proxy_address,
368        GovernanceClientCommands::AddTokensOnSui { .. } => unreachable!(),
369        GovernanceClientCommands::AddTokensOnEvm { .. } => config.eth_bridge_config_proxy_address,
370    }
371}

415impl LoadedBridgeCliConfig {
416    pub async fn load(cli_config: BridgeCliConfig) -> anyhow::Result<Self> {
417        if cli_config.eth_key_path.is_none() && cli_config.sui_key_path.is_none() {
418            return Err(anyhow!(
419                "At least one of `sui_key_path` or `eth_key_path` must be provided"
420            ));
421        }
422        let sui_key = if let Some(sui_key_path) = &cli_config.sui_key_path {
423            Some(read_key(sui_key_path, false)?)
424        } else {
425            None
426        };
427        let eth_key = if let Some(eth_key_path) = &cli_config.eth_key_path {
428            let eth_key = read_key(eth_key_path, true)?;
429            Some(eth_key)
430        } else {
431            None
432        };
433        let (eth_key, sui_key) = match (eth_key, sui_key) {
434            (None, Some(sui_key)) => {
435                if !matches!(sui_key, SuiKeyPair::Secp256k1(_)) {
436                    return Err(anyhow!("Eth key must be an ECDSA key"));
437                }
438                (sui_key.copy(), sui_key)
439            }
440            (Some(eth_key), None) => (eth_key.copy(), eth_key),
441            (Some(eth_key), Some(sui_key)) => (eth_key, sui_key),
442            (None, None) => unreachable!(),
443        };
444
445        let private_key_hex = Hex::encode(eth_key.to_bytes_no_flag());
446        let eth_signer_provider =
447            get_eth_signer_provider(&cli_config.eth_rpc_url, &private_key_hex)?;
448        let sui_bridge = EthSuiBridge::new(
449            cli_config.eth_bridge_proxy_address,
450            eth_signer_provider.clone(),
451        );
452        let eth_bridge_committee_proxy_address: EthAddress = sui_bridge.committee().call().await?;
453        let eth_bridge_limiter_proxy_address: EthAddress = sui_bridge.limiter().call().await?;
454        let eth_committee = EthBridgeCommittee::new(
455            eth_bridge_committee_proxy_address,
456            eth_signer_provider.clone(),
457        );
458        let eth_bridge_committee_proxy_address: EthAddress = sui_bridge.committee().call().await?;
459        let eth_bridge_config_proxy_address: EthAddress = eth_committee.config().call().await?;
460
461        let eth_address = eth_signer_provider.default_signer_address();
462        let eth_chain_id = eth_signer_provider.get_chain_id().await?;
463        let sui_address = SuiAddress::from(&sui_key.public());
464        println!("Using Sui address: {:?}", sui_address);
465        println!("Using Eth address: {:?}", eth_address);
466        println!("Using Eth chain: {:?}", eth_chain_id);
467
468        Ok(Self {
469            sui_rpc_url: cli_config.sui_rpc_url,
470            eth_rpc_url: cli_config.eth_rpc_url,
471            eth_bridge_proxy_address: cli_config.eth_bridge_proxy_address,
472            eth_bridge_committee_proxy_address,
473            eth_bridge_limiter_proxy_address,
474            eth_bridge_config_proxy_address,
475            sui_key,
476            eth_signer_provider,
477        })
478    }
479}

486    pub async fn get_sui_account_info(
487        self: &LoadedBridgeCliConfig,
488    ) -> anyhow::Result<(SuiKeyPair, SuiAddress, ObjectRef)> {
489        let pubkey = self.sui_key.public();
490        let sui_client_address = SuiAddress::from(&pubkey);
491        let sui_sdk_client = SuiClientBuilder::default()
492            .build(self.sui_rpc_url.clone())
493            .await?;
494        let gases = sui_sdk_client
495            .coin_read_api()
496            .get_coins(sui_client_address, None, None, None)
497            .await?
498            .data;
499        // TODO: is 5 Sui a good number?
500        let gas = gases
501            .into_iter()
502            .find(|coin| coin.balance >= 5_000_000_000)
503            .ok_or(anyhow!(
504                "Did not find gas object with enough balance for {}",
505                sui_client_address
506            ))?;
507        println!("Using Gas object: {}", gas.coin_object_id);
508        Ok((self.sui_key.copy(), sui_client_address, gas.object_ref()))
509    }

543impl BridgeClientCommands {
544    pub async fn handle(
545        self,
546        config: &LoadedBridgeCliConfig,
547        sui_bridge_client: SuiBridgeClient,
548    ) -> anyhow::Result<()> {
549        match self {
550            BridgeClientCommands::DepositNativeEtherOnEth {
551                ether_amount,
552                target_chain,
553                sui_recipient_address,
554            } => {
555                let eth_sui_bridge = EthSuiBridge::new(
556                    config.eth_bridge_proxy_address,
557                    config.eth_signer_provider().clone(),
558                );
559                // Note: even with f64 there may still be loss of precision even there are a lot of 0s
560                let int_part = ether_amount.trunc() as u64;
561                let frac_part = ether_amount.fract();
562                let int_wei = U256::from(int_part) * U256::from(10).pow(U256::from(18));
563                let frac_wei = U256::from((frac_part * 1_000_000_000_000_000_000f64) as u64);
564                let amount = int_wei + frac_wei;
565                let eth_tx = eth_sui_bridge
566                    .bridgeETH(sui_recipient_address.to_vec().into(), target_chain)
567                    .value(amount);
568                let pending_tx = eth_tx.send().await?;
569                let tx_receipt = pending_tx.get_receipt().await?;
570                info!(
571                    "Deposited {ether_amount} Ethers to {:?} (target chain {target_chain}). Receipt: {:?}",
572                    sui_recipient_address, tx_receipt,
573                );
574                Ok(())
575            }
576            BridgeClientCommands::ClaimOnEth { seq_num, dry_run } => {
577                claim_on_eth(seq_num, config, sui_bridge_client, dry_run)
578                    .await
579                    .map_err(|e| anyhow!("{:?}", e))
580            }
581            BridgeClientCommands::DepositOnSui {
582                coin_object_id,
583                coin_type,
584                target_chain,
585                recipient_address,
586            } => {
587                let target_chain = BridgeChainId::try_from(target_chain).expect("Invalid chain id");
588                let coin_type = TypeTag::from_str(&coin_type).expect("Invalid coin type");
589                deposit_on_sui(
590                    coin_object_id,
591                    coin_type,
592                    target_chain,
593                    recipient_address,
594                    config,

596                )
597                .await
598            }

603async fn deposit_on_sui(
604    coin_object_id: ObjectID,
605    coin_type: TypeTag,
606    target_chain: BridgeChainId,
607    recipient_address: EthAddress,
608    config: &LoadedBridgeCliConfig,
609    sui_bridge_client: SuiBridgeClient,
610) -> anyhow::Result<()> {
611    let target_chain = target_chain as u8;
612    let sui_client = sui_bridge_client.jsonrpc_client();
613    let bridge_object_arg = sui_bridge_client
614        .get_mutable_bridge_object_arg_must_succeed()
615        .await;
616    let rgp = sui_client
617        .governance_api()
618        .get_reference_gas_price()
619        .await
620        .unwrap();
621    let sender = SuiAddress::from(&config.sui_key.public());
622    let gas_obj_ref = sui_client
623        .coin_read_api()
624        .select_coins(sender, None, 1_000_000_000, vec![])
625        .await?
626        .first()
627        .ok_or(anyhow!("No coin found for address {}", sender))?
628        .object_ref();
629    let coin_obj_ref = sui_client
630        .read_api()
631        .get_object_with_options(coin_object_id, SuiObjectDataOptions::default())
632        .await?
633        .data
634        .unwrap()
635        .object_ref();
636
637    let mut builder = ProgrammableTransactionBuilder::new();
638    let arg_target_chain = builder.pure(target_chain).unwrap();
639    let arg_target_address = builder.pure(recipient_address.as_slice()).unwrap();
640    let arg_token = builder
641        .obj(ObjectArg::ImmOrOwnedObject(coin_obj_ref))
642        .unwrap();
643    let arg_bridge = builder.obj(bridge_object_arg).unwrap();
644
645    builder.programmable_move_call(
646        BRIDGE_PACKAGE_ID,
647        BRIDGE_MODULE_NAME.to_owned(),
648        ident_str!("send_token").to_owned(),
649        vec![coin_type],
650        vec![arg_bridge, arg_target_chain, arg_target_address, arg_token],
651    );
652    let pt = builder.finish();
653    let tx_data =
654        TransactionData::new_programmable(sender, vec![gas_obj_ref], pt, 500_000_000, rgp);
655    let sig = Signature::new_secure(
656        &IntentMessage::new(Intent::sui_transaction(), tx_data.clone()),
657        &config.sui_key,
658    );
659    let signed_tx = Transaction::from_data(tx_data, vec![sig]);
660    let tx_digest = *signed_tx.digest();
661    info!(?tx_digest, "Sending deposit transction to Sui.");
662    let resp = sui_bridge_client
663        .execute_transaction_block_with_effects(signed_tx)
664        .await
665        .expect("Failed to execute transaction block");
666    match &resp.status {
667        sui_json_rpc_types::SuiExecutionStatus::Success => {
668            info!(
669                ?tx_digest,
670                "Deposit transaction succeeded. Events: {:?}", resp.events
671            );
672            Ok(())
673        }
674        sui_json_rpc_types::SuiExecutionStatus::Failure { error } => {
675            Err(anyhow!("Transaction {:?} failed: {:?}", tx_digest, error))
676        }
677    }
678}

680async fn claim_on_eth(
681    seq_num: u64,
682    config: &LoadedBridgeCliConfig,
683    sui_bridge_client: SuiBridgeClient,
684    dry_run: bool,
685) -> BridgeResult<()> {
686    let sui_chain_id = sui_bridge_client.get_bridge_summary().await?.chain_id;
687    let parsed_message = sui_bridge_client
688        .get_parsed_token_transfer_message(sui_chain_id, seq_num)
689        .await?;
690    if parsed_message.is_none() {
691        println!("No record found for seq_num: {seq_num}, chain id: {sui_chain_id}");
692        return Ok(());
693    }
694    let parsed_message = parsed_message.unwrap();
695    let message_version = parsed_message.message_version;
696    let sigs = sui_bridge_client
697        .get_token_transfer_action_onchain_signatures_until_success(sui_chain_id, seq_num)
698        .await;
699    if sigs.is_none() {
700        println!("No signatures found for seq_num: {seq_num}, chain id: {sui_chain_id}");
701        return Ok(());
702    }
703    let signatures = sigs
704        .unwrap()
705        .into_iter()
706        .map(|sig: Vec<u8>| Bytes::from(sig))
707        .collect::<Vec<_>>();
708
709    let eth_sui_bridge = EthSuiBridge::new(
710        config.eth_bridge_proxy_address,
711        Arc::new(config.eth_signer_provider().clone()),
712    );
713    let message = eth_sui_bridge::BridgeUtils::Message::from(parsed_message);
714    let tx = if message_version == TOKEN_TRANSFER_MESSAGE_VERSION_V2 {
715        eth_sui_bridge
716            .transferBridgedTokensWithSignaturesV2(signatures, message)
717            .into_transaction_request()
718    } else {
719        eth_sui_bridge
720            .transferBridgedTokensWithSignatures(signatures, message)
721            .into_transaction_request()
722    };
723
724    if dry_run {
725        let resp = config.eth_signer_provider.estimate_gas(tx).await?;
726        println!(
727            "Sui to Eth bridge transfer claim dry run result: {:?}",
728            resp
729        );
730    } else {
731        let eth_claim_tx_receipt = config
732            .eth_signer_provider
733            .send_transaction(tx)
734            .await?
735            .get_receipt()
736            .await?;
737        println!(
738            "Sui to Eth bridge transfer claimed: {:?}",
739            eth_claim_tx_receipt
740        );
741    }
742    Ok(())
743}
