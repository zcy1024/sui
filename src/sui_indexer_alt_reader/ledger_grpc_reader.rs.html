35/// A reader backed by gRPC LedgerService (sui-kv-rpc).
36///
37/// This connects to archival service that implements the same LedgerService gRPC interface
38/// as fullnode, but is backed by Bigtable for serving historical data.
39#[derive(Clone)]
40pub struct LedgerGrpcReader(pub(crate) LedgerServiceClient<Channel>);
41
42impl LedgerGrpcArgs {
43    pub fn statement_timeout(&self) -> Option<std::time::Duration> {
44        self.ledger_grpc_statement_timeout_ms
45            .map(Duration::from_millis)
46    }
47}
48
49impl LedgerGrpcReader {
50    pub async fn new(uri: Uri, args: LedgerGrpcArgs) -> anyhow::Result<Self> {
51        let tls_config = ClientTlsConfig::new().with_native_roots();
52
53        let mut endpoint = Channel::builder(uri);
54        if let Some(timeout) = args.statement_timeout() {
55            endpoint = endpoint.timeout(timeout);
56        }
57        let channel = endpoint.tls_config(tls_config)?.connect_lazy();
58
59        let client = LedgerServiceClient::new(channel.clone());
60        Ok(Self(client))
61    }
62
63    pub fn as_data_loader(&self) -> DataLoader<Self> {
64        DataLoader::new(self.clone(), tokio::spawn)
65    }
66
67    pub async fn checkpoint_watermark(&self) -> anyhow::Result<CheckpointSummary> {
68        use prost_types::FieldMask;
69        use sui_rpc::field::FieldMaskUtil;
70        use sui_rpc::proto::sui::rpc::v2::GetCheckpointRequest;
71
72        let request =
73            GetCheckpointRequest::default().with_read_mask(FieldMask::from_paths(["summary.bcs"]));
74
75        let response = self
76            .0
77            .clone()
78            .get_checkpoint(request)
79            .await
80            .context("Failed to get latest checkpoint")?;
81
82        let checkpoint = response
83            .into_inner()
84            .checkpoint
85            .context("No checkpoint returned")?;
86
87        checkpoint
88            .summary
89            .as_ref()
90            .and_then(|s| s.bcs.as_ref())
91            .context("Missing summary.bcs")?
92            .deserialize()
93            .context("Failed to deserialize checkpoint summary")
94    }
95}
