75impl QueryOptions {
76    pub(crate) fn from_proto(
77        request: Option<&ProtoQueryOptions>,
78        default_limit_items: u32,
79        max_limit_items: u32,
80        query_type: QueryType,
81        filter: Option<&impl Message>,
82    ) -> Result<Self, RpcError> {
83        let limit_items = request
84            .and_then(|options| options.limit_items)
85            .unwrap_or(default_limit_items)
86            .clamp(1, max_limit_items) as usize;
87
88        let ordering = match request.map(|options| options.ordering) {
89            None | Some(ORDERING_ASCENDING) => Ordering::Ascending,
90            Some(ORDERING_DESCENDING) => Ordering::Descending,
91            Some(_) => {
92                return Err(FieldViolation::new("options.ordering")
93                    .with_description("invalid ordering")
94                    .with_reason(ErrorReason::FieldInvalid)
95                    .into());
96            }
97        };
98
99        let scope_digest = scope_digest(filter);
100        let after = parse_cursor(
101            "options.after",
102            request.and_then(|options| options.after.as_ref()),
103            query_type,
104            scope_digest,
105        )?;
106        let before = parse_cursor(
107            "options.before",
108            request.and_then(|options| options.before.as_ref()),
109            query_type,
110            scope_digest,
111        )?;
112
113        Ok(Self {
114            query_type,
115            limit_items,
116            ordering,
117            after,
118            before,
119            scope_digest,
120        })
121    }
122
123    pub(crate) fn scan_direction(&self) -> ScanDirection {
124        match self.ordering {
125            Ordering::Ascending => ScanDirection::Ascending,
126            Ordering::Descending => ScanDirection::Descending,
127        }
128    }
129
130    pub(crate) fn is_ascending(&self) -> bool {
131        matches!(self.ordering, Ordering::Ascending)
132    }

134    pub(crate) fn apply_cursor_bounds(&self, resolved: ResolvedRange) -> ResolvedRange {
135        if resolved.is_empty() {
136            return resolved;
137        }
138
139        let mut start = resolved.range.start;
140        let mut end = resolved.range.end;
141        let mut end_checkpoint = resolved.end_checkpoint;
142        let mut end_position = resolved.end_position;
143        let mut end_reason = resolved.end_reason;
144        let mut cursor_terminal = None;
145
146        if let Some(cursor) = &self.after {
147            let Some(after) = cursor.after_position_start() else {
148                return ResolvedRange::empty_at(
149                    cursor.checkpoint,
150                    cursor.position,
151                    QueryEndReason::CursorBound,
152                );
153            };
154            if after >= start {
155                start = after;
156                if matches!(self.ordering, Ordering::Descending) || after >= end {
157                    cursor_terminal = Some((cursor.checkpoint, after));
158                }
159                if matches!(self.ordering, Ordering::Descending) {
160                    end_checkpoint = cursor.checkpoint;
161                    end_position = after;
162                    end_reason = QueryEndReason::CursorBound;
163                }
164            }
165        }
166
167        if let Some(cursor) = &self.before
168            && cursor.position <= end
169        {
170            end = cursor.position;
171            if matches!(self.ordering, Ordering::Ascending) || cursor.position <= start {
172                cursor_terminal = Some((cursor.checkpoint, cursor.position));
173            }
174            if matches!(self.ordering, Ordering::Ascending) {
175                end_checkpoint = cursor.checkpoint;
176                end_position = cursor.position;
177                end_reason = QueryEndReason::CursorBound;
178            }
179        }
180
181        if start >= end {
182            if let Some((checkpoint, position)) = cursor_terminal {
183                end_checkpoint = checkpoint;
184                end_position = position;
185            }
186            if self.after.is_some() || self.before.is_some() {
187                end_reason = QueryEndReason::CursorBound;
188            }
189            ResolvedRange::empty_at(end_checkpoint, end_position, end_reason)
190        } else {
191            ResolvedRange {
192                range: start..end,
193                end_checkpoint,
194                end_position,
195                end_reason,
196            }
197        }
198    }

200    pub(crate) fn cursor_for_item(&self, checkpoint: u64, position: u64) -> Bytes {
201        self.encode_cursor(CursorKind::Item, checkpoint, position)
202    }

204    pub(crate) fn cursor_for_boundary(&self, checkpoint: u64, position: u64) -> Bytes {
205        self.encode_cursor(CursorKind::Boundary, checkpoint, position)
206    }

208    fn encode_cursor(&self, kind: CursorKind, checkpoint: u64, position: u64) -> Bytes {
209        encode_cursor(CursorToken {

211            kind,
212            checkpoint,
213            position,
214            scope_digest: self.scope_digest,
215        })

219impl ResolvedCheckpointRange {
220    pub(crate) fn empty_at(checkpoint: u64, reason: QueryEndReason) -> Self {
221        Self {
222            range: checkpoint..checkpoint,
223            end_reason: reason,
224        }
225    }

227    pub(crate) fn is_empty(&self) -> bool {
228        self.range.is_empty()
229    }
230
231    pub(crate) fn terminal_checkpoint(&self, ordering: Ordering) -> u64 {
232        match ordering {
233            Ordering::Ascending => self.range.end,
234            Ordering::Descending => self.range.start,
235        }
236    }
237
238    pub(crate) fn with_range(self, range: Range<u64>, ordering: Ordering) -> ResolvedRange {
239        let end_position = match ordering {
240            Ordering::Ascending => range.end,
241            Ordering::Descending => range.start,
242        };
243        ResolvedRange {
244            range,
245            end_checkpoint: self.terminal_checkpoint(ordering),
246            end_position,
247            end_reason: self.end_reason,
248        }
249    }

275impl CheckpointRange {
276    pub(crate) fn from_request(
277        start_checkpoint: Option<u64>,
278        end_checkpoint: Option<u64>,
279        checkpoint_hi_exclusive: u64,
280    ) -> Result<Self, RpcError> {
281        let start = start_checkpoint.unwrap_or(0);
282        if let Some(end) = end_checkpoint
283            && end < start
284        {
285            return Err(FieldViolation::new("end_checkpoint")
286                .with_description(
287                    "end_checkpoint must be greater than or equal to start_checkpoint",
288                )
289                .with_reason(ErrorReason::FieldInvalid)
290                .into());
291        }
292
293        let requested_end = end_checkpoint.unwrap_or(checkpoint_hi_exclusive);
294        let high_reason = if end_checkpoint.is_none() || requested_end > checkpoint_hi_exclusive {
295            QueryEndReason::LedgerTip
296        } else {
297            QueryEndReason::CheckpointBound
298        };
299        let end = requested_end.min(checkpoint_hi_exclusive);
300
301        Ok(Self {
302            start,
303            end,
304            high_reason,
305            indexed_tip: checkpoint_hi_exclusive,
306        })
307    }
308
309    pub(crate) fn resolve(self, options: &QueryOptions) -> ResolvedCheckpointRange {
310        let mut start = self.start;
311        let mut end = self.end;
312        let mut low_reason = QueryEndReason::CheckpointBound;
313        let mut high_reason = self.high_reason;
314        let mut cursor_bound = false;
315
316        if let Some(cursor) = &options.after
317            && cursor.checkpoint >= start
318        {
319            start = cursor.checkpoint;
320            cursor_bound = true;
321            if matches!(options.ordering, Ordering::Descending) {
322                low_reason = QueryEndReason::CursorBound;
323            }
324        }
325
326        if let Some(cursor) = &options.before
327            && let Some(upper) = cursor.before_checkpoint_end()
328            && upper <= end
329        {
330            end = upper;
331            cursor_bound = true;
332            if matches!(options.ordering, Ordering::Ascending) {
333                high_reason = QueryEndReason::CursorBound;
334            }
335        }
336
337        if start >= self.indexed_tip {
338            return ResolvedCheckpointRange::empty_at(self.indexed_tip, QueryEndReason::LedgerTip);
339        }
340
341        if start >= end {
342            let reason = if cursor_bound {
343                QueryEndReason::CursorBound
344            } else {
345                match options.ordering {
346                    Ordering::Ascending => high_reason,
347                    Ordering::Descending => low_reason,
348                }
349            };
350            let checkpoint = match options.ordering {
351                Ordering::Ascending => end,
352                Ordering::Descending => start,
353            };
354            return ResolvedCheckpointRange::empty_at(checkpoint, reason);
355        }
356
357        match options.ordering {
358            Ordering::Ascending => {
359                let scan_end = end.min(start.saturating_add(MAX_CHECKPOINT_SCAN_WIDTH));
360                let reason = if scan_end < end {
361                    QueryEndReason::ScanLimit
362                } else {
363                    high_reason
364                };
365                ResolvedCheckpointRange {
366                    range: start..scan_end,
367                    end_reason: reason,
368                }
369            }
370            Ordering::Descending => {
371                let scan_start = start.max(end.saturating_sub(MAX_CHECKPOINT_SCAN_WIDTH));
372                let reason = if scan_start > start {
373                    QueryEndReason::ScanLimit
374                } else {
375                    low_reason
376                };
377                ResolvedCheckpointRange {
378                    range: scan_start..end,
379                    end_reason: reason,
380                }
381            }
382        }
383    }
384}

386#[derive(Clone, Debug, Eq, PartialEq, serde::Serialize, serde::Deserialize)]
387struct CursorToken {
388    query_type: QueryType,
389    kind: CursorKind,
390    checkpoint: u64,
391    position: u64,
392    scope_digest: [u8; 32],
393}
394
395impl CursorToken {
396    fn validate(&self, query_type: QueryType, scope_digest: [u8; 32]) -> bool {
397        self.query_type == query_type && self.scope_digest == scope_digest
398    }
399
400    fn after_position_start(&self) -> Option<u64> {
401        match self.kind {
402            CursorKind::Item => self.position.checked_add(1),
403            CursorKind::Boundary => Some(self.position),
404        }
405    }
406
407    fn before_checkpoint_end(&self) -> Option<u64> {
408        match self.kind {
409            CursorKind::Item => self.checkpoint.checked_add(1),
410            CursorKind::Boundary => Some(self.checkpoint),
411        }
412    }
413}
414
415fn parse_cursor(
416    field: &'static str,
417    cursor: Option<&Bytes>,
418    query_type: QueryType,
419    scope_digest: [u8; 32],
420) -> Result<Option<CursorToken>, RpcError> {
421    cursor
422        .map(|cursor| decode_cursor(field, cursor))
423        .transpose()?
424        .map(|token| {
425            if token.validate(query_type, scope_digest) {
426                Ok(token)
427            } else {
428                Err(invalid_cursor(field, "invalid cursor"))
429            }
430        })
431        .transpose()
432}
433
434fn decode_cursor(field: &'static str, cursor: &[u8]) -> Result<CursorToken, RpcError> {
435    bcs::from_bytes(cursor).map_err(|_| invalid_cursor(field, "invalid cursor"))
436}
437
438fn encode_cursor(cursor: CursorToken) -> Bytes {
439    bcs::to_bytes(&cursor).unwrap().into()
440}
441
442// This digest is a cursor-scope guard, not a portable canonical protobuf hash.
443// We intentionally hash the server-side prost value after tonic/prost has
444// decoded the request and dropped unknown fields, then re-encode it with our
445// generated serializer. Protobuf wire bytes are not canonical, so do not
446// replace this with hashing raw request bytes. If cursors need to remain
447// compatible across proto/codegen/schema changes, replace this with a versioned
448// canonical digest over the validated internal query representation.
449fn scope_digest<F: Message>(filter: Option<&F>) -> [u8; 32] {
450    let mut hasher = DefaultHash::default();
451    hash_optional_message(&mut hasher, filter);
452    hasher.finalize().digest
453}
454
455fn hash_optional_message<M: Message>(hasher: &mut DefaultHash, message: Option<&M>) {
456    match message {
457        None => hasher.update([0]),
458        Some(message) => {
459            hasher.update([1]);
460            let bytes = message.encode_to_vec();
461            let len = u32::try_from(bytes.len()).expect("scan scope part should fit in u32");
462            hasher.update(len.to_be_bytes());
463            hasher.update(bytes);
464        }
465    }
466}
467
468fn invalid_cursor(field: &'static str, description: impl Into<String>) -> RpcError {
469    FieldViolation::new(field)
470        .with_description(description)
471        .with_reason(ErrorReason::FieldInvalid)
472        .into()
473}
