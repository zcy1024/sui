66#[derive(Clone)]
67pub(crate) struct KvRpcMetrics {
68    bigtable_limiter: Arc<BigTableLimiterMetrics>,
69    response_render_latency_ms: HistogramVec,
70    stream_item_yield_wait_ms: HistogramVec,
71}
72
73impl KvRpcMetrics {
74    fn new(registry: &Registry) -> Arc<Self> {
75        Arc::new(Self {
76            bigtable_limiter: BigTableLimiterMetrics::new(registry),
77            response_render_latency_ms: register_histogram_vec_with_registry!(
78                "kv_rpc_response_render_latency_ms",
79                "Wall time spent rendering one v2alpha response item.",
80                &["method"],
81                prometheus::exponential_buckets(0.01, 2.0, 18).unwrap(),
82                registry,
83            )
84            .unwrap(),
85            stream_item_yield_wait_ms: register_histogram_vec_with_registry!(
86                "kv_rpc_stream_item_yield_wait_ms",
87                "Wall time from yielding one v2alpha response item until the stream is polled again.",
88                &["method"],
89                prometheus::exponential_buckets(0.01, 2.0, 18).unwrap(),
90                registry,
91            )
92            .unwrap(),
93        })
94    }
95
96    fn observe_response_render(&self, method: &'static str, elapsed: std::time::Duration) {
97        self.response_render_latency_ms
98            .with_label_values(&[method])
99            .observe(elapsed.as_secs_f64() * 1000.0);
100    }
101
102    fn observe_stream_item_yield_wait(&self, method: &'static str, elapsed: std::time::Duration) {
103        self.stream_item_yield_wait_ms
104            .with_label_values(&[method])
105            .observe(elapsed.as_secs_f64() * 1000.0);
106    }
107}

126    pub metrics_registry: Option<Registry>,
127    pub enable_reflection: bool,

131impl KvRpcServer {
132    pub async fn new(
133        instance_id: String,
134        project_id: Option<String>,
135        app_profile_id: Option<String>,
136        checkpoint_bucket: Option<String>,
137        channel_timeout: Option<Duration>,
138        server_version: Option<ServerVersion>,
139        registry: &Registry,
140        credentials_path: Option<String>,
141        pool_config: PoolConfig,
142        service_info_watermark_pipelines: Vec<&'static str>,
143        concurrency: ConcurrencyConfig,
144    ) -> anyhow::Result<Self> {
145        concurrency.validate()?;
146        let mut client = BigTableClient::new_remote_with_credentials(
147            instance_id,
148            project_id,
149            false,
150            channel_timeout,
151            None,
152            "sui-kv-rpc".to_string(),
153            Some(registry),
154            app_profile_id,
155            pool_config,
156            credentials_path,
157        )
158        .await?;
159        let genesis = client
160            .get_checkpoints(&[0])
161            .await?
162            .pop()
163            .expect("failed to fetch genesis checkpoint from the KV store");
164        let summary = genesis.summary.expect("genesis checkpoint missing summary");
165        let chain_id = ChainIdentifier::from(summary.digest());
166        let metrics = KvRpcMetrics::new(registry);
167        Self::init(
168            client,
169            chain_id,
170            server_version,
171            checkpoint_bucket,
172            service_info_watermark_pipelines,
173            metrics,
174            concurrency,
175        )
176    }

178    /// Construct a KvRpcServer backed by a local BigTable emulator.
179    pub async fn new_local(
180        host: String,
181        instance_id: String,
182        server_version: Option<ServerVersion>,
183        checkpoint_bucket: Option<String>,
184    ) -> anyhow::Result<Self> {
185        let client = BigTableClient::new_local(host, instance_id).await?;
186        // Emulator/test path: metrics are inert (no scrape endpoint), but the
187        // request-scoped BigTable wrapper still expects a handle.
188        let metrics = KvRpcMetrics::new(&Registry::default());
189        Self::init(
190            client,
191            ChainIdentifier::from(sui_types::digests::CheckpointDigest::default()),
192            server_version,
193            checkpoint_bucket,
194            default_service_info_watermark_pipelines(false),
195            metrics,
196            ConcurrencyConfig::default(),
197        )
198    }
199
200    fn init(
201        client: BigTableClient,
202        chain_id: ChainIdentifier,
203        server_version: Option<ServerVersion>,
204        checkpoint_bucket: Option<String>,
205        service_info_watermark_pipelines: Vec<&'static str>,
206        metrics: Arc<KvRpcMetrics>,
207        concurrency: ConcurrencyConfig,
208    ) -> anyhow::Result<Self> {
209        ensure!(
210            !service_info_watermark_pipelines.is_empty(),
211            "at least one service info watermark pipeline must be configured"
212        );
213        concurrency.validate()?;
214
215        let cache = Arc::new(RwLock::new(None));
216
217        let package_store: Arc<dyn PackageStore> = Arc::new(PackageStoreWithLruCache::new(
218            BigTablePackageStore::new(client.clone()),
219        ));
220        let package_resolver = Arc::new(Resolver::new(package_store));
221
222        let server = Self {
223            chain_id,
224            client,
225            server_version,
226            checkpoint_bucket,
227            service_info_watermark_pipelines,
228            cache,
229            package_resolver,
230            metrics,
231            concurrency,
232        };
233
234        let server_clone = server.clone();
235        tokio::spawn(async move {
236            loop {
237                match v2::get_service_info(
238                    server_clone.client.clone(),
239                    server_clone.chain_id,
240                    server_clone.server_version.clone(),
241                    &server_clone.service_info_watermark_pipelines,
242                )
243                .await
244                {
245                    Ok(info) => {
246                        let mut cache = server_clone.cache.write().await;
247                        *cache = Some(info);
248                    }
249                    Err(e) => error!("Failed to update service info cache: {:?}", e),
250                }
251                sleep(Duration::from_millis(10)).await;
252            }
253        });
254
255        Ok(server)
256    }

258    /// Start this server as a tonic gRPC service on the given address.
259    /// Returns a `Service` handle for lifecycle management.
260    pub async fn start_service(
261        self,
262        listen_address: SocketAddr,
263        config: ServerConfig,
264    ) -> anyhow::Result<sui_futures::service::Service> {
265        use mysten_network::callback::CallbackLayer;
266        use sui_rpc_api::{
267            RpcMetrics, RpcMetricsMakeCallbackHandler, grpc_method_paths_from_file_descriptor_sets,
268        };
269
270        let mut builder = Server::builder();
271
272        if let Some(identity) = config.tls_identity {
273            builder = builder.tls_config(ServerTlsConfig::new().identity(identity))?;
274        }
275
276        // Single source of truth for every encoded FileDescriptorSet that
277        // backs a gRPC service mounted below. Consumed by both the
278        // reflection services and the metrics allowlist so they cannot drift
279        // out of sync.
280        let enable_experimental_query_apis = config.enable_experimental_query_apis;
281        let mut file_descriptor_sets: Vec<&'static [u8]> = vec![
282            sui_rpc_api::proto::google::protobuf::FILE_DESCRIPTOR_SET,
283            sui_rpc_api::proto::google::rpc::FILE_DESCRIPTOR_SET,
284            sui_rpc::proto::sui::rpc::v2::FILE_DESCRIPTOR_SET,
285        ];
286        if enable_experimental_query_apis {
287            file_descriptor_sets.push(sui_rpc::proto::sui::rpc::v2alpha::FILE_DESCRIPTOR_SET);
288        }
289
290        let registry = config.metrics_registry.unwrap_or_default();
291        let grpc_method_allowlist = Arc::new(grpc_method_paths_from_file_descriptor_sets(
292            &file_descriptor_sets,
293        )?);
294        let mut router = builder
295            .layer(CallbackLayer::new(
296                RpcMetricsMakeCallbackHandler::with_grpc_method_allowlist(
297                    Arc::new(RpcMetrics::new(&registry)),
298                    grpc_method_allowlist,
299                ),
300            ))
301            .add_service(LedgerServiceServer::new(self.clone()));
302
303        if enable_experimental_query_apis {
304            router = router.add_service(KvLedgerServiceServer::new(self));
305        }
306
307        if config.enable_reflection {
308            let mut reflection_v1_builder = tonic_reflection::server::Builder::configure();
309            let mut reflection_v1alpha_builder = tonic_reflection::server::Builder::configure();
310            for &fds in &file_descriptor_sets {
311                reflection_v1_builder =
312                    reflection_v1_builder.register_encoded_file_descriptor_set(fds);
313                reflection_v1alpha_builder =
314                    reflection_v1alpha_builder.register_encoded_file_descriptor_set(fds);
315            }
316            router = router
317                .add_service(reflection_v1_builder.build_v1()?)
318                .add_service(reflection_v1alpha_builder.build_v1alpha()?);
319        }
320
321        let (tx, rx) = tokio::sync::oneshot::channel::<()>();
322        let server_future = router.serve_with_shutdown(listen_address, async {
323            let _ = rx.await;
324        });
325
326        let service = sui_futures::service::Service::new()
327            .spawn(async move {
328                server_future.await?;
329                Ok(())
330            })
331            .with_shutdown_signal(async move {
332                let _ = tx.send(());
333            });
334
335        Ok(service)
336    }
337}

339pub fn default_service_info_watermark_pipelines(
340    enable_experimental_query_apis: bool,
341) -> Vec<&'static str> {
342    let mut pipelines = DEFAULT_SERVICE_INFO_WATERMARK_PIPELINES.to_vec();
343    if enable_experimental_query_apis {
344        pipelines.extend_from_slice(&EXPERIMENTAL_QUERY_SERVICE_INFO_WATERMARK_PIPELINES);
345    }
346    pipelines
