70pub(crate) async fn list_checkpoints(
71    ctx: QueryContext,
72    request: ListCheckpointsRequest,
73) -> Result<ListCheckpointsStream, RpcError> {
74    let started = Instant::now();
75    let filtered = request.filter.is_some();
76    let client: BigTableClient = ctx.client().clone();
77    let checkpoint_hi_exclusive = ctx.checkpoint_hi_exclusive();
78
79    let checkpoint_range = CheckpointRange::from_request(
80        request.start_checkpoint,
81        request.end_checkpoint,
82        checkpoint_hi_exclusive,
83    )?;
84    let read_mask = {
85        let read_mask = request
86            .read_mask
87            .unwrap_or_else(|| FieldMask::from_str(READ_MASK_DEFAULT));
88        read_mask.validate::<Checkpoint>().map_err(|path| {
89            FieldViolation::new("read_mask")
90                .with_description(format!("invalid read_mask path: {path}"))
91                .with_reason(ErrorReason::FieldInvalid)
92        })?;
93        FieldMaskTree::from(read_mask)
94    };
95    let options = QueryOptions::from_proto(
96        request.options.as_ref(),
97        DEFAULT_LIMIT_ITEMS,
98        MAX_LIMIT_ITEMS,
99        QueryType::Checkpoints,
100        request.filter.as_ref(),
101    )?;
102    let limit_items = options.limit_items;
103    let ordering = options.ordering;
104
105    let cp_range = async { Ok::<_, RpcError>(resolve_cp_range(checkpoint_range, &options)) }
106        .instrument(debug_span!("resolve_cp_range"))
107        .await?;
108    let end_reason = cp_range.end_reason;
109    let end_cursor = cp_range.end_cursor(&options);
110    let cp_range = cp_range.range;
111
112    if cp_range.is_empty() {
113        info!(
114            filtered,
115            limit_items,
116            ?ordering,
117            emitted = 0usize,
118            elapsed_ms = started.elapsed().as_millis(),
119            "list_checkpoints: empty range"
120        );
121        return Ok(stream::once(async move { Ok(end_response(end_reason, end_cursor)) }).boxed());
122    }
123
124    let needs_full = needs_transactions_or_objects(&read_mask);
125    let cp_columns: Arc<[&'static str]> = list_checkpoint_columns(&read_mask, needs_full).into();
126
127    let request_bigtable_concurrency = ctx.request_bigtable_concurrency();
128
129    // Stage A: discover cp_seq values for the requested response. Filtered
130    // requests use bitmap-eval, bounded by `max_bitmap_filter_literals`;
131    // unfiltered requests produce the seq range directly (cheap, no IO).
132    let seq_stream: BoxStream<'static, Result<u64, RpcError>> = if let Some(filter) =
133        &request.filter
134    {
135        filtered_checkpoint_seq_stream(&ctx, filter, cp_range.clone(), limit_items, options.clone())
136            .await?
137    } else {
138        range_stream(cp_range.clone(), &options)
139    };
140    let seq_stream = seq_stream.take(limit_items).boxed();
141
142    // Stage B: cp_seq -> (cp_seq, CheckpointData). One multi_get per chunk
143    // against the checkpoints table; each chunk is drained before rows are
144    // emitted downstream.
145    let cp_data_stream = pipelined_chunks(seq_stream, CHUNK_MAX, request_bigtable_concurrency, {
146        let client = client.clone();
147        let columns = cp_columns.clone();
148        move |seqs| fetch_checkpoint_data(client.clone(), columns.clone(), seqs)
149    });
150
151    // Fast path: read_mask doesn't request transactions or objects → render
152    // directly from CheckpointData via the existing `checkpoint_to_response`
153    // (with `checkpoint_bucket = None`, the GCS branch is a no-op).
154    if !needs_full {
155        return Ok(async_stream::try_stream! {
156            futures::pin_mut!(cp_data_stream);
157            let mut emitted = 0usize;
158            let mut last_cursor = None;
159            while let Some((cp_seq, cp_data)) = cp_data_stream.try_next().await? {
160                emitted += 1;
161                let message =
162                    crate::v2::get_checkpoint::checkpoint_to_response(cp_data, &read_mask, None)
163                        .await?;
164                let cursor = options.cursor_for_item(cp_seq, cp_seq);
165                last_cursor = Some(cursor.clone());
166                yield response_for(cursor, message);
167            }
168            let (reason, cursor) = query_end(emitted, limit_items, last_cursor, end_reason, end_cursor);
169            yield end_response(reason, cursor);
170            info!(
171                filtered,
172                limit_items,
173                ?ordering,
174                emitted,
175                elapsed_ms = started.elapsed().as_millis(),
176                "list_checkpoints: done (summary only)"
177            );
178        }
179        .boxed());
180    }
181
182    // Heavy path: needs transactions and/or objects.
183    let tx_columns: Arc<[&'static str]> = list_transactions_columns(&read_mask).into();
184    let needs_objects = read_mask.contains(Checkpoint::OBJECTS_FIELD);
185
186    // Stage C: (cp_seq, CheckpointData) -> + Vec<TransactionData>. Batched
187    // across the chunk: gather ALL tx_digests across all cps in the chunk
188    // into one multi_get, route results back per-cp, then emit in input
189    // cp_seq order after the chunk drains.
190    let cp_with_txs_stream =
191        pipelined_chunks(cp_data_stream, CHUNK_MAX, request_bigtable_concurrency, {
192            let client = client.clone();
193            let columns = tx_columns.clone();
194            move |items| fetch_transactions_for_cps(client.clone(), columns.clone(), items)
195        });
196
197    // Stage D: + ObjectMap. Per-cp object refs are precomputed, then
198    // `pipelined_keyed_batches` packs consecutive cps into batches whose
199    // deduped key union fits within CHUNK_MAX (see comment on the
200    // constant). The helper splits the per-batch fetch result back out
201    // per cp — `render_full_checkpoint` builds an `ObjectSet` by
202    // iterating the whole map, so each cp must see only its own keys.
203    let object_cache = ObjectCache::new(Arc::new(BigTableObjectFetcher::new(client.clone())));
204    let cp_full_stream: BoxStream<'static, Result<ResolvedCp, RpcError>> = if needs_objects {
205        let cp_with_keys = cp_with_txs_stream
206            .map_ok(|(cp_seq, cp_data, txs)| {
207                let keys: Vec<ObjectKey> = txs
208                    .iter()
209                    .flat_map(compute_object_keys)
210                    .collect::<BTreeSet<_>>()
211                    .into_iter()
212                    .collect();
213                ((cp_seq, cp_data, txs), keys)
214            })
215            .boxed();
216        pipelined_keyed_batches(
217            cp_with_keys,
218            CHUNK_MAX,
219            CHUNK_MAX,
220            request_bigtable_concurrency,
221            {
222                let object_cache = object_cache.clone();
223                move |keys| {
224                    let object_cache = object_cache.clone();
225                    async move { object_cache.get_many(keys).await }
226                }
227            },
228        )
229        .map_ok(|((cp_seq, cp_data, txs), objects)| (cp_seq, cp_data, txs, objects))
230        .boxed()
231    } else {
232        let empty: ObjectMap = Arc::new(HashMap::new());
233        cp_with_txs_stream
234            .map_ok(move |(cp_seq, cp_data, txs)| (cp_seq, cp_data, txs, empty.clone()))
235            .boxed()
236    };
237
238    // Stage E: sync render — build full_checkpoint_content::Checkpoint and
239    // merge into the proto Checkpoint (CPU-only, no further IO).
240    Ok(async_stream::try_stream! {
241        // Hold the cache for the lifetime of the response stream so in-flight
242        // object dispatches are aborted if the consumer drops the stream.
243        let _object_cache = object_cache;
244        futures::pin_mut!(cp_full_stream);
245        let mut emitted = 0usize;
246        let mut last_cursor = None;
247        while let Some(item) = cp_full_stream.try_next().await? {
248            let (cp_seq, cp_data, txs, objects) = item;
249            let cursor = options.cursor_for_item(cp_seq, cp_seq);
250            let response = render_full_checkpoint(
251                (cp_seq, cp_data, txs, objects),
252                &read_mask,
253                cursor.clone(),
254            )?;
255            emitted += 1;
256            last_cursor = Some(cursor);
257            yield response;
258        }
259        let (reason, cursor) = query_end(emitted, limit_items, last_cursor, end_reason, end_cursor);
260        yield end_response(reason, cursor);
261        info!(
262            filtered,
263            limit_items,
264            ?ordering,
265            emitted,
266            elapsed_ms = started.elapsed().as_millis(),
267            "list_checkpoints: done"
268        );
269    }
270    .boxed())
271}

273async fn fetch_checkpoint_data(
274    client: BigTableClient,
275    columns: Arc<[&'static str]>,
276    seqs: Vec<u64>,
277) -> Result<BoxStream<'static, Result<(u64, CheckpointData), RpcError>>, RpcError> {
278    if seqs.is_empty() {
279        return Ok(stream::empty().boxed());
280    }
281    let column_filter = BigTableClient::column_filter(&columns);
282    let keys: Vec<Vec<u8>> = seqs
283        .iter()
284        .copied()
285        .map(tables::checkpoints::encode_key)
286        .collect();
287    let rows = client
288        .multi_get_stream(
289            tables::checkpoints::NAME,
290            keys,
291            Some(column_filter),
292            stage::CHECKPOINTS,
293        )
294        .await?;
295
296    Ok(async_stream::try_stream! {
297        let mut emitter: InputOrderEmitter<u64, (u64, CheckpointData)> =
298            InputOrderEmitter::new(seqs);
299        futures::pin_mut!(rows);
300        while let Some(row) = rows.next().await {
301            let (key, cells) = row.map_err(RpcError::from)?;
302            let seq = decode_checkpoint_row_key(&key)?;
303            let checkpoint = tables::checkpoints::decode(&cells)?;
304            for v in emitter.push(
305                seq,
306                (seq, checkpoint),
307                "list_checkpoints: checkpoint lookup",
308            )? {
309                yield v;
310            }
311        }
312        for v in emitter.finish("list_checkpoints: missing selected checkpoint row")? {
313            yield v;
314        }
315    }
316    .boxed())
317}

319async fn fetch_transactions_for_cps(
320    client: BigTableClient,
321    columns: Arc<[&'static str]>,
322    items: Vec<(u64, CheckpointData)>,
323) -> Result<BoxStream<'static, Result<CpWithTxs, RpcError>>, RpcError> {
324    if items.is_empty() {
325        return Ok(stream::empty().boxed());
326    }
327
328    let mut input_order: Vec<u64> = Vec::with_capacity(items.len());
329    let mut cp_data_by_seq: HashMap<u64, CheckpointData> = HashMap::with_capacity(items.len());
330    let mut expected_count: HashMap<u64, usize> = HashMap::with_capacity(items.len());
331    let mut digest_to_cp: HashMap<TransactionDigest, u64> = HashMap::new();
332    let mut txs_by_seq: HashMap<u64, Vec<TransactionData>> = HashMap::with_capacity(items.len());
333    let mut flat_digests: Vec<TransactionDigest> = Vec::new();
334
335    for (cp_seq, cp_data) in items {
336        let contents = cp_data.contents.as_ref().ok_or_else(|| {
337            RpcError::new(
338                tonic::Code::Internal,
339                format!("checkpoint {cp_seq} contents column missing"),
340            )
341        })?;
342        let cp_digests: Vec<_> = contents.iter().map(|d| d.transaction).collect();
343        expected_count.insert(cp_seq, cp_digests.len());
344        txs_by_seq.insert(cp_seq, Vec::with_capacity(cp_digests.len()));
345        for digest in &cp_digests {
346            digest_to_cp.insert(*digest, cp_seq);
347        }
348        flat_digests.extend(cp_digests);
349        input_order.push(cp_seq);
350        cp_data_by_seq.insert(cp_seq, cp_data);
351    }
352
353    // Empty checkpoints (zero transactions) generate no BigTable rows, so
354    // pre-collect them and release through the emitter before draining the
355    // tx stream — otherwise they'd never get emitted.
356    let empty_cps: Vec<u64> = input_order
357        .iter()
358        .copied()
359        .filter(|cp_seq| expected_count[cp_seq] == 0)
360        .collect();
361
362    // CRITICAL: never call `get_transactions_stream` with an empty digest
363    // list. `multi_get_stream` builds a ReadRowsRequest with
364    // `rows_limit = keys.len()`, and BigTable interprets `rows_limit = 0`
365    // as "no limit" — i.e. a full transactions-table scan. When every cp
366    // in this chunk is empty, fall back to an empty stream and let the
367    // pre-emit loop alone drive emission.
368    let tx_stream: BoxStream<'static, Result<(TransactionDigest, TransactionData), RpcError>> =
369        if flat_digests.is_empty() {
370            stream::empty().boxed()
371        } else {
372            let column_filter = BigTableClient::column_filter(&columns);
373            client
374                .get_transactions_stream(flat_digests, Some(column_filter))
375                .await?
376                .map(|r| r.map_err(RpcError::from))
377                .boxed()
378        };
379
380    Ok(async_stream::try_stream! {
381        let mut emitter: InputOrderEmitter<u64, CpWithTxs> = InputOrderEmitter::new(input_order);
382        for cp_seq in empty_cps {
383            let cp_data = cp_data_by_seq.remove(&cp_seq).expect("cp_data entry present");
384            for v in emitter.push(
385                cp_seq,
386                (cp_seq, cp_data, Vec::new()),
387                "list_checkpoints: checkpoint transaction lookup",
388            )? {
389                yield v;
390            }
391        }
392        futures::pin_mut!(tx_stream);
393        while let Some(row) = tx_stream.next().await {
394            let (digest, tx) = row?;
395            let cp_seq = digest_to_cp.remove(&digest).ok_or_else(|| {
396                RpcError::new(
397                    tonic::Code::Internal,
398                    format!("list_checkpoints: unexpected transaction body row {digest}"),
399                )
400            })?;
401            let cp_txs = txs_by_seq
402                .get_mut(&cp_seq)
403                .expect("txs_by_seq entry present");
404            cp_txs.push(tx);
405            if cp_txs.len() == expected_count[&cp_seq] {
406                let txs = txs_by_seq.remove(&cp_seq).expect("txs_by_seq entry");
407                let cp_data = cp_data_by_seq
408                    .remove(&cp_seq)
409                    .expect("cp_data entry present");
410                for v in emitter.push(
411                    cp_seq,
412                    (cp_seq, cp_data, txs),
413                    "list_checkpoints: checkpoint transaction lookup",
414                )? {
415                    yield v;
416                }
417            }
418        }
419        // Defensive: if BigTable returned fewer rows than requested, surface
420        // the missing digests as an internal error rather than emit a
421        // partial checkpoint downstream. cp_data_by_seq still containing
422        // entries means at least one cp's tx set was incomplete.
423        if !cp_data_by_seq.is_empty() {
424            // Build a per-cp report (got vs expected) to make production
425            // triage tractable — without it the error message alone gives
426            // no signal about which cps or how big the gap was.
427            let mut incomplete: Vec<(u64, usize, usize)> = cp_data_by_seq
428                .keys()
429                .map(|cp_seq| {
430                    let got = txs_by_seq.get(cp_seq).map(|v| v.len()).unwrap_or(0);
431                    let expected = expected_count.get(cp_seq).copied().unwrap_or(0);
432                    (*cp_seq, got, expected)
433                })
434                .collect();
435            incomplete.sort_unstable();
436            tracing::warn!(
437                incomplete_count = incomplete.len(),
438                ?incomplete,
439                "list_checkpoints: BigTable returned fewer transactions than requested (cp_seq, got, expected)"
440            );
441            Err(RpcError::new(
442                tonic::Code::Internal,
443                format!(
444                    "list_checkpoints: BigTable returned fewer transactions than requested for {} checkpoint(s)",
445                    incomplete.len()
446                ),
447            ))?;
448        }
449        for v in emitter.finish("list_checkpoints: missing selected checkpoint transactions")? {
450            yield v;
451        }
452    }
453    .boxed())
454}

456fn render_full_checkpoint(
457    item: ResolvedCp,
458    read_mask: &FieldMaskTree,
459    cursor: Bytes,
460) -> Result<ListCheckpointsResponse, RpcError> {
461    let (cp_seq, cp_data, txs, objects) = item;
462
463    let summary = cp_data.summary.ok_or_else(|| {
464        RpcError::new(
465            tonic::Code::Internal,
466            format!("checkpoint {cp_seq} summary column missing"),
467        )
468    })?;
469    let signatures = cp_data.signatures.ok_or_else(|| {
470        RpcError::new(
471            tonic::Code::Internal,
472            format!("checkpoint {cp_seq} signatures column missing"),
473        )
474    })?;
475    let contents = cp_data.contents.ok_or_else(|| {
476        RpcError::new(
477            tonic::Code::Internal,
478            format!("checkpoint {cp_seq} contents column missing"),
479        )
480    })?;
481
482    let executed_transactions = txs
483        .into_iter()
484        .map(|tx| {
485            let transaction = tx.transaction_data.ok_or_else(|| {
486                RpcError::new(
487                    tonic::Code::Internal,
488                    format!("transaction {} data column missing", tx.digest),
489                )
490            })?;
491            let effects = tx.effects.ok_or_else(|| {
492                RpcError::new(
493                    tonic::Code::Internal,
494                    format!("transaction {} effects column missing", tx.digest),
495                )
496            })?;
497            Ok::<_, RpcError>(FullExecutedTransaction {
498                transaction,
499                signatures: tx.signatures.unwrap_or_default(),
500                effects,
501                events: tx.events,
502                unchanged_loaded_runtime_objects: tx.unchanged_loaded_runtime_objects,
503            })
504        })
505        .collect::<Result<Vec<_>, _>>()?;
506
507    let mut object_set = ObjectSet::default();
508    for (_, obj) in objects.iter() {
509        object_set.insert(obj.clone());
510    }
511
512    let full_checkpoint = FullCheckpoint {
513        summary: CertifiedCheckpointSummary::new_from_data_and_sig(summary, signatures),
514        contents,
515        transactions: executed_transactions,
516        object_set,
517    };
518
519    let mut message = Checkpoint::default();
520    message.merge(&full_checkpoint, read_mask);
521
522    Ok(response_for(cursor, message))
523}
524
525fn response_for(cursor: Bytes, message: Checkpoint) -> ListCheckpointsResponse {
526    let mut item = CheckpointItem::default();
527    item.cursor = Some(cursor);
528    item.checkpoint = Some(message);
529
530    let mut response = ListCheckpointsResponse::default();
531    response.response = Some(list_checkpoints_response::Response::Item(item));
532    response
533}
534
535fn end_response(reason: QueryEndReason, cursor: Bytes) -> ListCheckpointsResponse {
536    let mut end = QueryEnd::default();
537    end.cursor = Some(cursor);
538    end.reason = reason as i32;
539
540    let mut response = ListCheckpointsResponse::default();
541    response.response = Some(list_checkpoints_response::Response::End(end));
542    response
543}

545fn query_end(
546    emitted: usize,
547    limit_items: usize,
548    last_cursor: Option<Bytes>,
549    end_reason: QueryEndReason,
550    end_cursor: Bytes,
551) -> (QueryEndReason, Bytes) {
552    if emitted == limit_items {
553        (
554            QueryEndReason::ItemLimit,
555            last_cursor.expect("item-limit responses have a last cursor"),
556        )
557    } else {
558        (end_reason, end_cursor)
559    }
560}

562fn needs_transactions_or_objects(mask: &FieldMaskTree) -> bool {
563    mask.contains(Checkpoint::TRANSACTIONS_FIELD) || mask.contains(Checkpoint::OBJECTS_FIELD)
564}
565
566/// Columns to fetch from the checkpoints table. When transactions or
567/// objects are in the read mask we additionally need `signatures` (to
568/// reconstruct `CertifiedCheckpointSummary`) and `contents` (to enumerate
569/// each checkpoint's transaction digests).
570fn list_checkpoint_columns(mask: &FieldMaskTree, needs_full: bool) -> Vec<&'static str> {
571    let mut columns = checkpoint_columns(mask);
572    if needs_full {
573        if !columns.contains(&tables::checkpoints::col::CONTENTS) {
574            columns.push(tables::checkpoints::col::CONTENTS);
575        }
576        if !columns.contains(&tables::checkpoints::col::SIGNATURES) {
577            columns.push(tables::checkpoints::col::SIGNATURES);
578        }
579    }
580    columns
581}
582
583/// Columns to fetch from the transactions table for the heavy path. The
584/// merge target is `full_checkpoint_content::ExecutedTransaction`, whose
585/// `transaction: TransactionData` and `effects: TransactionEffects` fields
586/// are non-`Option` — even when the read mask only asks for
587/// `transactions.digest`, the merge reads `source.transaction.digest()` to
588/// produce the response, so we must always have data + effects available.
589/// Object resolution likewise needs data + effects + unchanged_loaded to
590/// derive `compute_object_keys`. Optional source fields (signatures,
591/// events, unchanged_loaded) are gated by the mask.
592fn list_transactions_columns(mask: &FieldMaskTree) -> Vec<&'static str> {
593    let mut columns = if let Some(submask) = mask.subtree(Checkpoint::TRANSACTIONS_FIELD.name) {
594        transaction_columns(&submask)
595    } else {
596        // Baseline metadata columns even if the proto `transactions` field
597        // isn't requested; we still need the rows to compute object keys.
598        vec![
599            tables::transactions::col::CHECKPOINT_NUMBER,
600            tables::transactions::col::TIMESTAMP,
601        ]
602    };
603    // Required to construct the merge target faithfully — see fn doc.
604    for col in [
605        tables::transactions::col::DATA,
606        tables::transactions::col::EFFECTS,
607    ] {
608        if !columns.contains(&col) {
609            columns.push(col);
610        }
611    }
612    if mask.contains(Checkpoint::OBJECTS_FIELD)
613        && !columns.contains(&tables::transactions::col::UNCHANGED_LOADED)
614    {
615        columns.push(tables::transactions::col::UNCHANGED_LOADED);
616    }
617    columns
618}

620async fn filtered_checkpoint_seq_stream(
621    ctx: &QueryContext,
622    filter: &sui_rpc::proto::sui::rpc::v2alpha::TransactionFilter,
623    cp_range: std::ops::Range<u64>,
624    limit: usize,
625    options: QueryOptions,
626) -> Result<BoxStream<'static, Result<u64, RpcError>>, RpcError> {
627    if limit == 0 || cp_range.is_empty() {
628        return Ok(stream::empty().boxed());
629    }
630
631    let client = ctx.client();
632    let query = ctx.transaction_filter_query(filter)?;
633    let tx_range = client.checkpoint_to_tx_range(cp_range).await?;
634    if tx_range.is_empty() {
635        return Ok(stream::empty().boxed());
636    }
637
638    let tx_seq_stream = client
639        .eval_bitmap_query_stream(
640            query,
641            tx_range,
642            BitmapIndexSpec::tx(),
643            options.scan_direction(),
644        )
645        .map_err(RpcError::from);
646    let fetch_client: BigTableClient = client.clone();
647
648    Ok(async_stream::try_stream! {
649        futures::pin_mut!(tx_seq_stream);
650        let mut tx_seq_chunk = Vec::with_capacity(CHUNK_MAX);
651        let mut last_cp_seq = None;
652        let mut emitted = 0usize;
653
654        while emitted < limit {
655            tx_seq_chunk.clear();
656            while tx_seq_chunk.len() < CHUNK_MAX {
657                match tx_seq_stream.try_next().await? {
658                    Some(tx_seq) => tx_seq_chunk.push(tx_seq),
659                    None => break,
660                }
661            }
662
663            if tx_seq_chunk.is_empty() {
664                break;
665            }
666
667            let mut tx_checkpoints = fetch_client.resolve_tx_checkpoints(&tx_seq_chunk).await?;
668            if options.is_ascending() {
669                tx_checkpoints.sort_by_key(|(tx_seq, _)| *tx_seq);
670            } else {
671                tx_checkpoints.sort_by_key(|(tx_seq, _)| std::cmp::Reverse(*tx_seq));
672            }
673
674            for (_, cp_seq) in tx_checkpoints {
675                if last_cp_seq == Some(cp_seq) {
676                    continue;
677                }
678
679                last_cp_seq = Some(cp_seq);
680                emitted += 1;
681                yield cp_seq;
682
683                if emitted >= limit {
684                    break;
685                }
686            }
687        }
688    }
689    .boxed())
690}

692/// Determine the checkpoint_sequence_number scan window from the logical
693/// checkpoint bounds, indexed history, cursor bounds, and the per-request scan
694/// width.
695fn resolve_cp_range(checkpoint_range: CheckpointRange, options: &QueryOptions) -> ResolvedRange {
696    let cp_range = checkpoint_range.resolve(options);
697    let range = cp_range.range.clone();
698    options.apply_cursor_bounds(cp_range.with_range(range, options.ordering))
699}

701fn decode_checkpoint_row_key(key: &Bytes) -> Result<u64, RpcError> {
702    let bytes: [u8; 8] = key
703        .as_ref()
704        .try_into()
705        .map_err(|_| RpcError::new(tonic::Code::Internal, "invalid checkpoint row key length"))?;
706    Ok(u64::from_be_bytes(bytes))
707}

709fn range_stream(
710    range: std::ops::Range<u64>,
711    options: &QueryOptions,
712) -> BoxStream<'static, Result<u64, RpcError>> {
713    if options.is_ascending() {
714        stream::iter(range.map(Ok::<_, RpcError>)).boxed()
715    } else {
716        stream::iter(range.rev().map(Ok::<_, RpcError>)).boxed()
717    }
718}
