23pub(crate) struct EagerFundsWithdrawScheduler {
24    funds_read: Arc<dyn AccountFundsRead>,
25    inner_state: Arc<Mutex<InnerState>>,
26}
27
28struct InnerState {
29    /// For each address funds account that we have seen withdraws through `schedule_withdraws`,
30    /// we track the current state of that account, and only remove it from the map after
31    /// we have settled all withdraws for that account.
32    tracked_accounts: HashMap<AccumulatorObjId, AccountState>,
33    /// Tracks all the address funds accounts that have a withdraw transaction tracked,
34    /// mapping from the accumulator version that the withdraw transaction reads from.
35    /// If a withdraw transaction needs to withdraw from account O at version V,
36    /// we must process and settle that withdraw transaction whenever we settle all transactions
37    /// scheduled for version V.
38    pending_settlements: HashMap<SequenceNumber, BTreeSet<AccumulatorObjId>>,
39    /// The current version of the accumulator object.
40    accumulator_version: SequenceNumber,
41}
42
43struct AccountState {
44    account_id: AccumulatorObjId,
45    /// The amount of funds that has been reserved for this account, for each accumulator version.
46    /// This is tracked so that we could add them back to the account funds when we settle the withdraws.
47    reserved_funds: HashMap<SequenceNumber, u128>,
48    /// Withdraws that could not yet be scheduled due to insufficient funds, and
49    /// hence have not reserved any funds yet. We track them so that we could schedule them
50    /// anytime we may have sufficient funds.
51    pending_reservations: VecDeque<Arc<PendingWithdraw>>,
52    /// The lower bound of the current funds of this account.
53    /// It is the amount of guaranteed funds that we could withdraw from this account at this point.
54    /// This is maintained as the most recent settled funds, subtracted by the reserved funds.
55    funds_lower_bound: u128,
56}
57
58struct PendingWithdraw {
59    accumulator_version: SequenceNumber,
60    tx_digest: TransactionDigest,
61    sender: Mutex<Option<Sender<ScheduleResult>>>,
62    pending: Mutex<BTreeMap<AccumulatorObjId, u64>>,
63}
64
65enum TryReserveResult {
66    SufficientFunds,
67    InsufficientFunds,
68    Pending,
69}
70
71impl EagerFundsWithdrawScheduler {
72    pub fn new(
73        funds_read: Arc<dyn AccountFundsRead>,
74        starting_accumulator_version: SequenceNumber,
75    ) -> Arc<Self> {
76        Arc::new(Self {
77            funds_read,
78            inner_state: Arc::new(Mutex::new(InnerState {
79                tracked_accounts: HashMap::new(),
80                pending_settlements: HashMap::new(),
81                accumulator_version: starting_accumulator_version,
82            })),
83        })
84    }
85}

87#[async_trait::async_trait]
88impl FundsWithdrawSchedulerTrait for EagerFundsWithdrawScheduler {
89    #[instrument(level = "debug", skip_all, fields(withdraw_accumulator_version = ?withdraws.accumulator_version.value()))]
90    async fn schedule_withdraws(&self, withdraws: WithdrawReservations) {
91        let mut inner_state = self.inner_state.lock();
92        let cur_accumulator_version = inner_state.accumulator_version;
93        if withdraws.accumulator_version < cur_accumulator_version {
94            // This accumulator version is already settled.
95            // There is no need to schedule the withdraws.
96            withdraws.notify_skip_schedule();
97            return;
98        }
99
100        for (withdraw, sender) in withdraws.withdraws.into_iter().zip(withdraws.senders) {
101            let accounts = withdraw.reservations.keys().cloned().collect::<Vec<_>>();
102            inner_state
103                .pending_settlements
104                .entry(withdraws.accumulator_version)
105                .or_default()
106                .extend(&accounts);
107            let pending_withdraw =
108                PendingWithdraw::new(withdraws.accumulator_version, withdraw, sender);
109            for account_id in accounts {
110                let account_state = inner_state
111                    .tracked_accounts
112                    .entry(account_id)
113                    .or_insert_with(|| {
114                        // TODO: This will be doing a DF read while holding the state lock.
115                        // We may need to look at ways to make the DF reads non-blocking.
116                        // We also need to get rid of the need to read old versions of the account funds.
117                        AccountState::new(
118                            self.funds_read.as_ref(),
119                            account_id,
120                            cur_accumulator_version,
121                        )
122                    });
123                let has_blocking_reservations = !account_state.pending_reservations.is_empty();
124                let result = account_state.try_reserve(
125                    cur_accumulator_version,
126                    &pending_withdraw,
127                    has_blocking_reservations,
128                );
129                if matches!(result, TryReserveResult::Pending) {
130                    account_state
131                        .pending_reservations
132                        .push_back(pending_withdraw.clone());
133                }
134            }
135        }
136    }
137
138    async fn settle_funds(&self, settlement: FundsSettlement) {
139        let next_accumulator_version = settlement.next_accumulator_version;
140        let mut inner_state = self.inner_state.lock();
141        if next_accumulator_version <= inner_state.accumulator_version {
142            // This accumulator version is already settled.
143            // There is no need to settle the funds.
144            debug!(
145                next_accumulator_version =? next_accumulator_version.value(),
146                "Skipping settlement since it is already settled",
147            );
148            return;
149        }
150        assert_eq!(
151            next_accumulator_version,
152            inner_state.accumulator_version.next()
153        );
154        let cleanup_version = inner_state.accumulator_version;
155        inner_state.accumulator_version.increment();
156        debug!(
157            "Bumping accumulator version from {:?} to {:?}",
158            cleanup_version.value(),
159            next_accumulator_version.value(),
160        );
161
162        // A settlement bumps the accumulator version from `cleanup_version` to `last_settled_version`.
163        // We must processing the following types of accounts:
164        // 1. Accounts that had withdraws scheduled at version `cleanup_version`. These withdraws
165        // are now settled and we know their exact funds changes.
166        // 2. Accounts that had withdraws scheduled at version `last_settled_version`. These withdraws
167        // if not yet scheduled, can now be deterministically scheduled, since
168        // we have the final state before them.
169        // 3. Accounts that have funds changes through settle_funds, this can include
170        // both withdraws and deposits.
171        let mut affected_accounts = inner_state
172            .pending_settlements
173            .remove(&cleanup_version)
174            .unwrap_or_default();
175        // Since we just bumped the accumulator version, any withdraws scheduled that depend
176        // on this version can now be deterministically scheduled.
177        if let Some(current_version_accounts) = inner_state
178            .pending_settlements
179            .get(&next_accumulator_version)
180        {
181            affected_accounts.extend(current_version_accounts);
182        }
183        affected_accounts.extend(settlement.funds_changes.keys().cloned());
184
185        debug!(
186            "Processing withdraws affecting accounts: {:?}",
187            affected_accounts,
188        );
189        for object_id in affected_accounts {
190            let Some(account_state) = inner_state.tracked_accounts.get_mut(&object_id) else {
191                continue;
192            };
193            debug!(
194                account_id = ?object_id,
195                "Settling account",
196            );
197            let reserved = account_state
198                .reserved_funds
199                .remove(&cleanup_version)
200                .unwrap_or_default() as i128;
201            let settled = settlement
202                .funds_changes
203                .get(&object_id)
204                .copied()
205                .unwrap_or_default();
206            let net = u128::try_from(reserved.checked_add(settled).unwrap())
207                .expect("Withdraw amounts must be bounded by reservations");
208            account_state.funds_lower_bound += net;
209            debug!(
210                account_id = ?object_id,
211                "Reserved funds: {:?}, settled funds: {:?}, new min guaranteed funds: {:?}",
212                reserved, settled, account_state.funds_lower_bound,
213            );
214
215            while !account_state.pending_reservations.is_empty() {
216                let pending_withdraw = account_state.pending_reservations.pop_front().unwrap();
217                assert!(pending_withdraw.accumulator_version >= next_accumulator_version);
218
219                let result =
220                    account_state.try_reserve(next_accumulator_version, &pending_withdraw, false);
221                if matches!(result, TryReserveResult::Pending) {
222                    account_state
223                        .pending_reservations
224                        .push_front(pending_withdraw);
225                    break;
226                }
227            }
228
229            if in_test_configuration() {
230                account_state.debug_check_account_state_invariants(
231                    self.funds_read.as_ref(),
232                    next_accumulator_version,
233                );
234            }
235
236            if account_state.is_empty() {
237                debug!(
238                    account_id = ?object_id,
239                    "Removing account state since it is empty",
240                );
241                inner_state.tracked_accounts.remove(&object_id);
242            }
243            // TODO: Debug invariant check on account state and accumulator version match.
244        }
245        debug!("Settled accumulator version {:?}", next_accumulator_version);
246    }
247
248    fn close_epoch(&self) {
249        debug!("Closing epoch in EagerFundsWithdrawScheduler",);
250        let inner_state = self.inner_state.lock();
251        assert!(inner_state.pending_settlements.is_empty());
252        assert!(inner_state.tracked_accounts.is_empty());
253    }
254
255    #[cfg(test)]
256    fn get_current_accumulator_version(&self) -> SequenceNumber {
257        let inner_state = self.inner_state.lock();
258        inner_state.accumulator_version
259    }
260}

262impl AccountState {
263    fn new(
264        funds_read: &dyn AccountFundsRead,
265        account_id: AccumulatorObjId,
266        last_settled_version: SequenceNumber,
267    ) -> Self {
268        let funds = funds_read.get_account_amount(&account_id, last_settled_version);
269        debug!(
270            last_settled_version =? last_settled_version.value(),
271            account_id = ?account_id.inner(),
272            "New account state tracked with initial funds {:?}",
273            funds,
274        );
275        Self {
276            account_id,
277            reserved_funds: HashMap::new(),
278            pending_reservations: VecDeque::new(),
279            funds_lower_bound: funds,
280        }
281    }
282
283    #[instrument(
284        level = "debug",
285        skip_all,
286        fields(
287            cur_accumulator_version =? cur_accumulator_version.value(),
288            account_id = ?self.account_id.inner(),
289            tx_digest = ?pending_withdraw.tx_digest,
290            ?has_blocking_reservations
291        )
292    )]
293    fn try_reserve(
294        &mut self,
295        cur_accumulator_version: SequenceNumber,
296        pending_withdraw: &Arc<PendingWithdraw>,
297        has_blocking_reservations: bool,
298    ) -> TryReserveResult {
299        let to_reserve = pending_withdraw.pending_amount(&self.account_id);
300        debug!(
301            "Trying to reserve {}, min_guaranteed_funds: {}, pending_reservations size: {}",
302            to_reserve,
303            self.funds_lower_bound,
304            self.pending_reservations.len(),
305        );
306        let insufficient_funds = to_reserve > self.funds_lower_bound;
307        if insufficient_funds || has_blocking_reservations {
308            if cur_accumulator_version == pending_withdraw.accumulator_version {
309                assert!(!has_blocking_reservations);
310                pending_withdraw.notify_insufficient_funds();
311                TryReserveResult::InsufficientFunds
312            } else {
313                debug!("Adding to pending reservations since we cannot schedule it yet.");
314                TryReserveResult::Pending
315            }
316        } else {
317            self.commit_reservation(pending_withdraw);
318            TryReserveResult::SufficientFunds
319        }
320    }
321
322    fn commit_reservation(&mut self, pending_withdraw: &Arc<PendingWithdraw>) {
323        let mut pending = pending_withdraw.pending.lock();
324        let to_reserve = pending.remove(&self.account_id).unwrap() as u128;
325        assert!(self.funds_lower_bound >= to_reserve);
326        self.funds_lower_bound -= to_reserve;
327        *self
328            .reserved_funds
329            .entry(pending_withdraw.accumulator_version)
330            .or_default() += to_reserve;
331        debug!(
332            "Successfully reserved {} for account. New min guaranteed funds: {}",
333            to_reserve, self.funds_lower_bound
334        );
335        if pending.is_empty() {
336            debug!("Successfully reserved all accounts for withdraw transaction");
337            let sender = pending_withdraw.sender.lock().take().unwrap();
338            let _ = sender.send(ScheduleResult {
339                tx_digest: pending_withdraw.tx_digest,
340                status: ScheduleStatus::SufficientFunds,
341            });
342        }
343    }
344
345    fn is_empty(&self) -> bool {
346        self.reserved_funds.is_empty() && self.pending_reservations.is_empty()
347    }
348
349    fn debug_check_account_state_invariants(
350        &self,
351        funds_read: &dyn AccountFundsRead,
352        last_settled_version: SequenceNumber,
353    ) {
354        let total_reserved = self.reserved_funds.values().sum::<u128>();
355        let expected_funds = self.funds_lower_bound + total_reserved;
356        let actual_funds = funds_read.get_account_amount(&self.account_id, last_settled_version);
357        assert_eq!(expected_funds, actual_funds);
358    }
359}
360
361impl PendingWithdraw {
362    fn new(
363        accumulator_version: SequenceNumber,
364        withdraw: TxFundsWithdraw,
365        sender: Sender<ScheduleResult>,
366    ) -> Arc<Self> {
367        Arc::new(Self {
368            accumulator_version,
369            tx_digest: withdraw.tx_digest,
370            sender: Mutex::new(Some(sender)),
371            pending: Mutex::new(withdraw.reservations),
372        })
373    }
374
375    fn pending_amount(&self, account_id: &AccumulatorObjId) -> u128 {
376        self.pending.lock().get(account_id).copied().unwrap() as u128
377    }
378
379    fn notify_insufficient_funds(&self) {
380        let mut sender_guard = self.sender.lock();
381        // sender may be None because this pending withdraw may have multiple
382        // insufficient accounts, and when processing the first one, the sender
383        // is already taken.
384        if let Some(sender) = sender_guard.take() {
385            debug!("Insufficient funds for withdraw");
386            let _ = sender.send(ScheduleResult {
387                tx_digest: self.tx_digest,
388                status: ScheduleStatus::InsufficientFunds,
389            });
390        }
391    }
