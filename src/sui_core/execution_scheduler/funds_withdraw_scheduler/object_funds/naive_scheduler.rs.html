21#[derive(Clone)]
22pub(crate) struct NaiveObjectFundsWithdrawScheduler {
23    funds_read: Arc<dyn AccountFundsRead>,
24    inner: Arc<RwLock<Inner>>,
25    accumulator_version_sender: Arc<watch::Sender<SequenceNumber>>,
26    // We must keep a receiver alive to make sure sends go through and can update the last settled version.
27    accumulator_version_receiver: Arc<watch::Receiver<SequenceNumber>>,
28    epoch_ended: Arc<CancellationToken>,
29}
30
31struct Inner {
32    /// Unsettled withdraws in the current consensus commit (identified by the accumulator version).
33    /// This is cleared whenever we settle a new version.
34    /// We must track these because when we execute a transaction, the witdhraws are not immediately settled,
35    /// so we need to track them and check them again when we execute the next transaction from the same consensus commit.
36    unsettled_withdraws: BTreeMap<AccumulatorObjId, u128>,
37}
38
39impl NaiveObjectFundsWithdrawScheduler {
40    pub fn new(
41        funds_read: Arc<dyn AccountFundsRead>,
42        starting_accumulator_version: SequenceNumber,
43    ) -> Self {
44        let (accumulator_version_sender, accumulator_version_receiver) =
45            watch::channel(starting_accumulator_version);
46        Self {
47            funds_read,
48            inner: Arc::new(RwLock::new(Inner {
49                unsettled_withdraws: BTreeMap::new(),
50            })),
51            accumulator_version_sender: Arc::new(accumulator_version_sender),
52            accumulator_version_receiver: Arc::new(accumulator_version_receiver),
53            epoch_ended: Arc::new(CancellationToken::new()),
54        }
55    }
56
57    fn try_withdraw(&self, object_withdraws: &BTreeMap<AccumulatorObjId, u64>) -> bool {
58        for (obj_id, amount) in object_withdraws {
59            // It is safe to get the latest funds here because this function is called during execution,
60            // which means this transaction is not committed yet,
61            // so the settlement transaction at the end of the same consensus commit cannot have settled yet.
62            // That is, we must be blocked by this transaction in order to make progress.
63            let funds = self.funds_read.get_latest_account_amount(obj_id);
64            let unsettled_withdraw = self
65                .inner
66                .read()
67                .unsettled_withdraws
68                .get(obj_id)
69                .copied()
70                .unwrap_or_default();
71            assert!(funds >= unsettled_withdraw);
72            if funds - unsettled_withdraw < *amount as u128 {
73                return false;
74            }
75        }
76        let mut inner = self.inner.write();
77        for (obj_id, amount) in object_withdraws {
78            let entry = inner.unsettled_withdraws.entry(*obj_id).or_default();
79            *entry += *amount as u128;
80        }
81        true
82    }
83
84    fn return_insufficient_funds() -> ObjectFundsWithdrawStatus {
85        let (sender, receiver) = oneshot::channel();
86        // unwrap is safe because the receiver is defined right above.
87        sender.send(FundsWithdrawStatus::Insufficient).unwrap();
88        ObjectFundsWithdrawStatus::Pending(receiver)
89    }
90}
91
92impl ObjectFundsWithdrawSchedulerTrait for NaiveObjectFundsWithdrawScheduler {
93    fn schedule(
94        &self,
95        object_withdraws: BTreeMap<AccumulatorObjId, u64>,
96        accumulator_version: SequenceNumber,
97    ) -> ObjectFundsWithdrawStatus {
98        let last_settled_version = *self.accumulator_version_receiver.borrow();
99        // This function is called during execution, which means this transaction is not committed yet,
100        // so the settlement transaction at the end of the same consensus commit cannot have settled yet.
101        assert!(
102            accumulator_version >= last_settled_version,
103            "accumulator_version: {}, last_settled_version: {}",
104            accumulator_version,
105            last_settled_version
106        );
107        if accumulator_version == last_settled_version {
108            if self.try_withdraw(&object_withdraws) {
109                return ObjectFundsWithdrawStatus::SufficientFunds;
110            } else {
111                return Self::return_insufficient_funds();
112            }
113        }
114
115        // Spawn a task to wait for the last settled version to become accumulator_version,
116        // before we could check again.
117        let accumulator_version_sender = self.accumulator_version_sender.clone();
118        let epoch_cancel = self.epoch_ended.child_token();
119        let (sender, receiver) = oneshot::channel();
120        tokio::spawn(async move {
121            let mut version_receiver = accumulator_version_sender.subscribe();
122            tokio::select! {
123                res = version_receiver.wait_for(|v| *v >= accumulator_version) => {
124                    if res.is_err() {
125                        // This shouldn't happen, but just to be safe.
126                        tracing::error!("Accumulator version receiver channel closed while waiting for accumulator version");
127                        return;
128                    }
129                    // We notify the waiter that the funds are now deterministically known,
130                    // but we don't need to check here whether they are sufficient or not.
131                    // Next time during execution we will check again.
132                    let _ = sender.send(FundsWithdrawStatus::MaybeSufficient);
133                }
134                _ = epoch_cancel.cancelled() => {}
135            }
136        });
137        ObjectFundsWithdrawStatus::Pending(receiver)
138    }
139
140    fn settle_accumulator_version(&self, next_accumulator_version: SequenceNumber) {
141        let mut inner = self.inner.write();
142        // unwrap is safe because the scheduler always holds a reference to the receiver.
143        self.accumulator_version_sender
144            .send(next_accumulator_version)
145            .unwrap();
146        inner.unsettled_withdraws.clear();
147    }
148
149    fn close_epoch(&self) {
150        self.epoch_ended.cancel();
151    }
152
153    #[cfg(test)]
154    fn get_current_accumulator_version(&self) -> SequenceNumber {
155        *self.accumulator_version_receiver.borrow()
156    }
157}
