53const WAIT_FOR_EFFECTS_TIMEOUT: Duration = Duration::from_secs(10);

55const MAX_WAIT_FOR_EFFECTS_RETRY_DELAY: Duration = Duration::from_secs(2);

57/// Delay before starting a speculative get_full_effects request to a fallback validator.
58/// If the first validator hasn't responded within this time, we start a parallel request
59/// to another validator that has already acknowledged the effects.
60const GET_FULL_EFFECTS_FALLBACK_DELAY: Duration = Duration::from_millis(200);
61
62/// Result type for get_full_effects requests.
63/// The tuple contains (effects_digest, executed_data, fast_path) where fast_path
64/// boolean indicateswhether the transaction was executed via the fast path.
65type FullEffectsResult =
66    Result<(TransactionEffectsDigest, Box<ExecutedData>, bool), TransactionRequestError>;

77    #[instrument(level = "error", skip_all, err(level = "debug"))]
78    pub(crate) async fn get_certified_finalized_effects<A>(
79        &self,
80        authority_aggregator: &Arc<AuthorityAggregator<A>>,
81        client_monitor: &Arc<ValidatorClientMonitor<A>>,
82        tx_digest: Option<TransactionDigest>,
83        tx_type: TxType,
84        // This keeps track of the current target for getting full effects.
85        mut current_target: AuthorityName,
86        // Guaranteed to be not the Rejected variant.
87        submit_txn_result: SubmitTxResult,
88        options: &SubmitTransactionOptions,
89    ) -> Result<QuorumTransactionResponse, TransactionDriverError>
90    where
91        A: AuthorityAPI + Send + Sync + 'static + Clone,
92    {
93        // When consensus position is provided, wait for finalized and fastpath outputs at the validators' side.
94        // Otherwise, only wait for finalized effects.
95        // Skip the first attempt to get full effects if it is already provided.
96        let (consensus_position, full_effects) = match submit_txn_result {
97            SubmitTxResult::Submitted { consensus_position } => (Some(consensus_position), None),
98            SubmitTxResult::Executed {
99                effects_digest,
100                details,
101                fast_path,
102            } => match details {
103                Some(details) => (None, Some((effects_digest, details, fast_path))),
104                // Details should always be set in correct responses.
105                // But if it is not set, continuing to get full effects and certify the digest are still correct.
106                None => (None, None),
107            },
108            SubmitTxResult::Rejected { error } => {
109                return Err(TransactionDriverError::ClientInternal {
110                    error: format!(
111                        "Unexpected submission error in get_certified_finalized_effects(): {}",
112                        error
113                    ),
114                });
115            }
116        };
117
118        let mut retrier = RequestRetrier::new(
119            authority_aggregator,
120            client_monitor,
121            tx_type,
122            vec![],
123            vec![],
124        );
125        let ping_type = get_ping_type(&tx_digest, tx_type);
126
127        // Channel for wait_for_acknowledgments to notify which validators have acked.
128        // These validators are known to have executed the transaction, making them good
129        // fallback candidates for get_full_effects if the initial validator is slow.
130        // Bounded by committee size since each validator sends at most one ack.
131        let (acked_validators_tx, acked_validators_rx) =
132            channel(authority_aggregator.committee.num_members());
133
134        // Setting this to None at first because if the full effects are already provided,
135        // we do not need to record the latency. We track the time in this function instead of inside
136        // get_full_effects so that we could record differently depending on whether the result is byzantine.
137        let mut full_effects_start_time = None;
138        let (acknowledgments_result, (mut full_effects_result, returned_target)) = join!(
139            self.wait_for_acknowledgments(
140                authority_aggregator,
141                client_monitor,
142                tx_digest,
143                tx_type,
144                consensus_position,
145                options,
146                current_target,
147                acked_validators_tx,
148            ),
149            async {
150                // No need to send a full effects request if it is already provided.
151                if let Some(full_effects) = full_effects {
152                    // In this branch, current_target is the authority providing the full effects,
153                    // so it is consistent. This is not used though because current_target is
154                    // only used with failed full effects query.
155                    return (Ok(full_effects), current_target);
156                }
157                let (name, client) = retrier
158                    .next_target()
159                    .expect("there should be at least 1 target");
160                full_effects_start_time = Some(Instant::now());
161                self.get_full_effects_with_fallback(
162                    authority_aggregator,
163                    client,
164                    name,
165                    tx_digest,
166                    tx_type,
167                    consensus_position,
168                    options,
169                    acked_validators_rx,
170                )
171                .await
172            },
173        );
174        current_target = returned_target;
175
176        // If the consensus position got rejected, effects certification will see the failure and gather
177        // error messages to explain the rejection.
178        let certified_digest = acknowledgments_result?;
179
180        // Retry until there is a valid full effects that matches the certified digest, or all targets
181        // have been attempted.
182        loop {
183            let display_name = authority_aggregator.get_display_name(&current_target);
184            match full_effects_result {
185                Ok((effects_digest, executed_data, _fast_path)) => {
186                    if effects_digest != certified_digest {
187                        tracing::warn!(
188                            ?current_target,
189                            "Full effects digest mismatch ({} vs certified {})",
190                            effects_digest,
191                            certified_digest
192                        );
193                        // This validator is byzantine, record the error.
194                        client_monitor.record_interaction_result(OperationFeedback {
195                            authority_name: current_target,
196                            display_name,
197                            operation: OperationType::Effects,
198                            ping_type,
199                            result: Err(()),
200                        });
201                    } else {
202                        if let Some(start_time) = full_effects_start_time {
203                            let latency = start_time.elapsed();
204                            client_monitor.record_interaction_result(OperationFeedback {
205                                authority_name: current_target,
206                                display_name,
207                                operation: OperationType::Effects,
208                                ping_type,
209                                result: Ok(latency),
210                            });
211                        }
212                        return Ok(
213                            self.get_quorum_transaction_response(effects_digest, executed_data)
214                        );
215                    }
216                }
217                Err(e) => {
218                    tracing::debug!(?current_target, "Failed to get full effects: {e}");
219                    client_monitor.record_interaction_result(OperationFeedback {
220                        authority_name: current_target,
221                        display_name,
222                        operation: OperationType::Effects,
223                        ping_type,
224                        result: Err(()),
225                    });
226                    // This emits an error when retrier gathers enough (f+1) non-retriable effects errors,
227                    // but the error should not happen after effects certification unless there are software bugs
228                    // or > f malicious validators.
229                    retrier.add_error(current_target, e)?;
230                }
231            };
232
233            tokio::task::yield_now().await;
234
235            // Retry getting full effects from the next target.
236
237            // This emits an error when retrier has no targets available.
238            let (name, client) = retrier.next_target()?;
239            current_target = name;
240            full_effects_start_time = Some(Instant::now());
241            full_effects_result = self
242                .get_full_effects(client, tx_digest, tx_type, consensus_position, options)
243                .await;
244        }
245    }

247    #[instrument(level = "debug", skip_all, err(level = "debug"), fields(tx_digest = ?tx_digest, consensus_position = ?consensus_position, ret_effects_digest = tracing::field::Empty))]
248    async fn get_full_effects<A>(
249        &self,
250        client: Arc<SafeClient<A>>,
251        tx_digest: Option<TransactionDigest>,
252        tx_type: TxType,
253        consensus_position: Option<ConsensusPosition>,
254        options: &SubmitTransactionOptions,
255    ) -> FullEffectsResult
256    where
257        A: AuthorityAPI + Send + Sync + 'static + Clone,
258    {
259        let ping_type = get_ping_type(&tx_digest, tx_type);
260        let request = WaitForEffectsRequest {
261            transaction_digest: tx_digest,
262            consensus_position,
263            include_details: true,
264            ping_type,
265        };
266
267        match timeout(
268            WAIT_FOR_EFFECTS_TIMEOUT,
269            client.wait_for_effects(request, options.forwarded_client_addr),
270        )
271        .await
272        {
273            Ok(Ok(response)) => match response {
274                WaitForEffectsResponse::Executed {
275                    effects_digest,
276                    details,
277                    fast_path,
278                } => {
279                    if let Some(details) = details {
280                        tracing::Span::current()
281                            .record("ret_effects_digest", format!("{:?}", effects_digest));
282                        Ok((effects_digest, details, fast_path))
283                    } else {
284                        tracing::debug!("Execution data not found, retrying...");
285                        Err(TransactionRequestError::ValidatorInternal(
286                            "Execution data not found".to_string(),
287                        ))
288                    }
289                }
290                WaitForEffectsResponse::Rejected { error } => match error {
291                    Some(e) => Err(TransactionRequestError::RejectedAtValidator(e)),
292                    // Even though this response is not an error, returning an error which is required
293                    // by the function signature. This will get ignored by the caller as a retriable error.
294                    None => Err(TransactionRequestError::RejectedByConsensus),
295                },
296                WaitForEffectsResponse::Expired { epoch, round } => Err(
297                    TransactionRequestError::StatusExpired(epoch, round.unwrap_or(0)),
298                ),
299            },
300            Ok(Err(e)) => Err(TransactionRequestError::Aborted(e)),
301            Err(_) => Err(TransactionRequestError::TimedOutGettingFullEffectsAtValidator),
302        }
303    }

305    /// Gets full effects from a validator, with speculative fallback to other validators.
306    ///
307    /// If the initial validator doesn't respond within GET_FULL_EFFECTS_FALLBACK_DELAY,
308    /// we start parallel requests to validators that have already acknowledged the effects
309    /// (received via the acked_validators channel from wait_for_acknowledgments).
310    ///
311    /// This prevents slow validators from blocking the entire operation when faster
312    /// validators are available, while still preferring the initial validator if it responds quickly.
313    #[instrument(level = "debug", skip_all, fields(tx_digest = ?tx_digest, initial_validator = ?initial_target))]
314    async fn get_full_effects_with_fallback<A>(
315        &self,
316        authority_aggregator: &Arc<AuthorityAggregator<A>>,
317        initial_client: Arc<SafeClient<A>>,
318        initial_target: AuthorityName,
319        tx_digest: Option<TransactionDigest>,
320        tx_type: TxType,
321        consensus_position: Option<ConsensusPosition>,
322        options: &SubmitTransactionOptions,
323        mut acked_validators_rx: Receiver<AuthorityName>,
324    ) -> (FullEffectsResult, AuthorityName)
325    where
326        A: AuthorityAPI + Send + Sync + 'static + Clone,
327    {
328        let mut pending_requests: FuturesUnordered<
329            BoxFuture<'_, (AuthorityName, FullEffectsResult)>,
330        > = FuturesUnordered::new();
331
332        // Add initial request to the pending set alongside fallbacks for uniform handling
333        let initial_request = self.get_full_effects(
334            initial_client,
335            tx_digest,
336            tx_type,
337            consensus_position,
338            options,
339        );
340        pending_requests.push(Box::pin(
341            async move { (initial_target, initial_request.await) },
342        ));
343
344        let mut fallback_delay = tokio::time::interval(GET_FULL_EFFECTS_FALLBACK_DELAY);
345        fallback_delay.reset();
346
347        loop {
348            tokio::select! {
349                Some((validator, result)) = pending_requests.next() => {
350                    // Return as soon as any request (including fallback)completes - the caller handles retries for errors
351                    return (result, validator);
352                }
353
354                // After delay, try to start a fallback request to an acked validator
355                _ = fallback_delay.tick() => {
356                    // Drain all available acked validators and pick one we haven't tried
357                    while let Ok(acked_validator) = acked_validators_rx.try_recv() {
358                        // We send ack requests to all validators, so skip if the acked validator was the initial target
359                        if acked_validator == initial_target {
360                            continue;
361                        }
362
363                        let Some(client) = authority_aggregator.authority_clients.get(&acked_validator) else {
364                            continue;
365                        };
366
367                        tracing::debug!(
368                            ?acked_validator,
369                            "Starting fallback get_full_effects request"
370                        );
371
372                        let fut = self.get_full_effects(
373                            client.clone(),
374                            tx_digest,
375                            tx_type,
376                            consensus_position,
377                            options,
378                        );
379
380                        pending_requests.push(Box::pin(async move {
381                            (acked_validator, fut.await)
382                        }));
383
384                        // Only start one fallback per interval
385                        break;
386                    }
387                }
388            }
389        }
390    }

392    #[instrument(level = "debug", skip_all, err(level = "debug"), ret, fields(consensus_position = ?consensus_position))]
393    async fn wait_for_acknowledgments<A>(
394        &self,
395        authority_aggregator: &Arc<AuthorityAggregator<A>>,
396        client_monitor: &Arc<ValidatorClientMonitor<A>>,
397        tx_digest: Option<TransactionDigest>,
398        tx_type: TxType,
399        consensus_position: Option<ConsensusPosition>,
400        options: &SubmitTransactionOptions,
401        submitted_tx_to_validator: AuthorityName,
402        acked_validators_tx: Sender<AuthorityName>,
403    ) -> Result<TransactionEffectsDigest, TransactionDriverError>
404    where
405        A: AuthorityAPI + Send + Sync + 'static + Clone,
406    {
407        let ping_type = get_ping_type(&tx_digest, tx_type);
408        let ping_label = if tx_digest.is_none() { "true" } else { "false" };
409        self.metrics
410            .certified_effects_ack_attempts
411            .with_label_values(&[tx_type.as_str(), ping_label])
412            .inc();
413        let timer = tokio::time::Instant::now();
414        let clients = authority_aggregator
415            .authority_clients
416            .iter()
417            .collect::<Vec<_>>();
418        let committee = authority_aggregator.committee.clone();
419        let raw_request = RawWaitForEffectsRequest::try_from(WaitForEffectsRequest {
420            transaction_digest: tx_digest,
421            consensus_position,
422            include_details: false,
423            ping_type,
424        })
425        .unwrap();
426
427        // Broadcast requests for digest acknowledgments against all validators.
428        let mut futures = FuturesUnordered::new();
429        for (name, client) in clients {
430            let client = client.clone();
431            let name = *name;
432            let display_name = authority_aggregator.get_display_name(&name);
433
434            let raw_request = raw_request.clone();
435            let future = async move {
436                match timeout(
437                    WAIT_FOR_EFFECTS_TIMEOUT,
438                    self.wait_for_acknowledgment_rpc(
439                        name,
440                        display_name.clone(),
441                        &client,
442                        client_monitor,
443                        &raw_request,
444                        options,
445                    ),
446                )
447                .await
448                {
449                    Ok(result) => (name, result),
450                    Err(_) => {
451                        client_monitor.record_interaction_result(OperationFeedback {
452                            authority_name: name,
453                            display_name,
454                            operation: OperationType::Effects,
455                            ping_type,
456                            result: Err(()),
457                        });
458                        (name, Err(SuiErrorKind::TimeoutError.into()))
459                    }
460                }
461            };
462
463            futures.push(future);
464        }
465
466        let mut effects_digest_aggregators: BTreeMap<
467            TransactionEffectsDigest,
468            StatusAggregator<()>,
469        > = BTreeMap::new();
470        // Collect responses from validators which observed the transaction getting rejected,
471        // and rejected the transaction with errors non-retriable with new transaction submissions.
472        let mut non_retriable_errors_aggregator =
473            StatusAggregator::<TransactionRequestError>::new(committee.clone());
474        // Collect responses from validators which observed the transaction getting rejected,
475        // and rejected the transaction with errors retriable with new transaction submissions.
476        let mut retriable_errors_aggregator =
477            StatusAggregator::<TransactionRequestError>::new(committee.clone());
478        // Collect responses from validators which observed the transaction getting rejected,
479        // but do not have a local reason to reject the transaction. The validator could have
480        // accepted the transaction during voting, or the reason has been lost.
481        let mut reason_not_found_aggregator = StatusAggregator::<()>::new(committee.clone());
482        // Collect responses from validators which observed the transaction getting executed using fast path.
483        let mut fast_path_aggregator = StatusAggregator::<()>::new(committee.clone());
484
485        // Every validator returns at most one WaitForEffectsResponse.
486        while let Some((name, response)) = futures.next().await {
487            match response {
488                Ok(WaitForEffectsResponse::Executed {
489                    effects_digest,
490                    details: _,
491                    fast_path,
492                }) => {
493                    // Notify that this validator has successfully executed the transaction.
494                    // This allows get_full_effects_with_fallback to use this validator as a
495                    // fallback if the initial validator is slow.
496                    // Using try_send since the channel is bounded by committee size and we don't
497                    // want to block - if the channel is somehow full, it's fine to skip.
498                    let _ = acked_validators_tx.try_send(name);
499
500                    if fast_path {
501                        if tx_type != TxType::SingleWriter {
502                            tracing::warn!(
503                                "Fast path is only supported for single writer transactions, name={name}"
504                            );
505                        } else {
506                            fast_path_aggregator.insert(name, ());
507                        }
508                    }
509
510                    let aggregator = effects_digest_aggregators
511                        .entry(effects_digest)
512                        .or_insert_with(|| StatusAggregator::<()>::new(committee.clone()));
513                    aggregator.insert(name, ());
514
515                    if aggregator.reached_quorum_threshold() {
516                        let quorum_weight = aggregator.total_votes();
517                        for (other_digest, other_aggregator) in effects_digest_aggregators {
518                            if other_digest != effects_digest && other_aggregator.total_votes() > 0
519                            {
520                                tracing::warn!(
521                                    ?name,
522                                    "Effects digest inconsistency detected: quorum digest {effects_digest:?} (weight {quorum_weight}), other digest {other_digest:?} (weight {})",
523                                    other_aggregator.total_votes()
524                                );
525                                self.metrics.effects_digest_mismatches.inc();
526                            }
527                        }
528                        // Record success and latency
529                        self.metrics
530                            .certified_effects_ack_successes
531                            .with_label_values(&[tx_type.as_str(), ping_label])
532                            .inc();
533                        self.metrics
534                            .certified_effects_ack_latency
535                            .with_label_values(&[tx_type.as_str(), ping_label])
536                            .observe(timer.elapsed().as_secs_f64());
537
538                        if fast_path_aggregator.reached_quorum_threshold() {
539                            // get the display name of the validator that the transaction has been submitted to
540                            let display_name =
541                                authority_aggregator.get_display_name(&submitted_tx_to_validator);
542
543                            self.metrics
544                                .transaction_fastpath_acked
545                                .with_label_values(&[&display_name, ping_label])
546                                .inc();
547                        }
548
549                        return Ok(effects_digest);
550                    }
551                }
552                Ok(WaitForEffectsResponse::Rejected { error }) => {
553                    if let Some(e) = error {
554                        tracing::trace!(name = ?name.concise(), "Rejected at validator: {:?}", e);
555                        let error = TransactionRequestError::RejectedAtValidator(e);
556                        if error.is_submission_retriable() {
557                            retriable_errors_aggregator.insert(name, error);
558                        } else {
559                            non_retriable_errors_aggregator.insert(name, error);
560                        }
561                    } else {
562                        tracing::trace!(name = ?name.concise(), "Not found at validator");
563                        reason_not_found_aggregator.insert(name, ());
564                    }
565                    self.metrics
566                        .rejection_acks
567                        .with_label_values(&[tx_type.as_str(), ping_label])
568                        .inc();
569                }
570                Ok(WaitForEffectsResponse::Expired { epoch, round }) => {
571                    let error = TransactionRequestError::StatusExpired(epoch, round.unwrap_or(0));
572                    // Expired status is submission retriable.
573                    retriable_errors_aggregator.insert(name, error);
574                    self.metrics
575                        .expiration_acks
576                        .with_label_values(&[tx_type.as_str(), ping_label])
577                        .inc();
578                }
579                Err(error) => {
580                    let error = TransactionRequestError::Aborted(error);
581                    if error.is_submission_retriable() {
582                        retriable_errors_aggregator.insert(name, error);
583                    } else {
584                        non_retriable_errors_aggregator.insert(name, error);
585                    }
586                }
587            };
588
589            // Adding vote up between different StatusAggregators without de-duplication is ok,
590            // because each authority only returns one response.
591            let executed_weight: u64 = effects_digest_aggregators
592                .values()
593                .map(|agg| agg.total_votes())
594                .sum();
595            let total_weight = executed_weight
596                + reason_not_found_aggregator.total_votes()
597                + non_retriable_errors_aggregator.total_votes()
598                + retriable_errors_aggregator.total_votes();
599            let remaining_weight = committee.total_votes() - total_weight;
600
601            // Wait for a quorum of responses, to not summarize the responses too early.
602            // If neither of the branches can exit the loop, the loop will eventually terminate when responses are
603            // gathered from all validators. The time is bounded by WAIT_FOR_EFFECTS_TIMEOUT.
604            //
605            // The most important goal here is to aggregate enough useful errors to be actionable.
606            // Breaking the loop at the earliest possible time is not the goal.
607            if total_weight >= committee.quorum_threshold() {
608                // Try returning non-retriable aggregated error first.
609                if non_retriable_errors_aggregator.total_votes() >= committee.validity_threshold() {
610                    return Err(TransactionDriverError::RejectedByValidators {
611                        submission_non_retriable_errors: aggregate_request_errors(
612                            non_retriable_errors_aggregator.status_by_authority(),
613                        ),
614                        submission_retriable_errors: aggregate_request_errors(
615                            retriable_errors_aggregator.status_by_authority(),
616                        ),
617                    });
618                }
619                // Return a retriable aggregated error only if it becomes impossible to gather enough non-retriable errors.
620                // reason_not_found_aggregator is excluded here intentionally because it does not contain a useful error message.
621                if non_retriable_errors_aggregator.total_votes() + remaining_weight
622                    < committee.validity_threshold()
623                    && retriable_errors_aggregator.total_votes()
624                        + non_retriable_errors_aggregator.total_votes()
625                        >= committee.validity_threshold()
626                {
627                    let mut observed_effects_digests =
628                        Vec::<(TransactionEffectsDigest, Vec<AuthorityName>, StakeUnit)>::new();
629                    for (effects_digest, aggregator) in effects_digest_aggregators {
630                        observed_effects_digests.push((
631                            effects_digest,
632                            aggregator.authorities(),
633                            aggregator.total_votes(),
634                        ));
635                    }
636                    return Err(TransactionDriverError::Aborted {
637                        submission_non_retriable_errors: aggregate_request_errors(
638                            non_retriable_errors_aggregator.status_by_authority(),
639                        ),
640                        submission_retriable_errors: aggregate_request_errors(
641                            retriable_errors_aggregator.status_by_authority(),
642                        ),
643                        observed_effects_digests: AggregatedEffectsDigests {
644                            digests: observed_effects_digests,
645                        },
646                    });
647                }
648            }
649        }
650
651        // At this point, no effects digest has reached quorum. But failed responses do not reach
652        // validity threshold either.
653        let retriable_weight =
654            retriable_errors_aggregator.total_votes() + reason_not_found_aggregator.total_votes();
655        // Whether the transaction is retriable regardless of known effects.
656        let mut submission_retriable = retriable_weight >= committee.quorum_threshold();
657        let mut observed_effects_digests =
658            Vec::<(TransactionEffectsDigest, Vec<AuthorityName>, StakeUnit)>::new();
659        for (effects_digest, aggregator) in effects_digest_aggregators {
660            // This effects digest can still get certified, so the transaction is retriable.
661            if aggregator.total_votes() + retriable_weight >= committee.quorum_threshold() {
662                submission_retriable = true;
663            }
664            observed_effects_digests.push((
665                effects_digest,
666                aggregator.authorities(),
667                aggregator.total_votes(),
668            ));
669        }
670        if submission_retriable {
671            Err(TransactionDriverError::Aborted {
672                submission_non_retriable_errors: aggregate_request_errors(
673                    non_retriable_errors_aggregator.status_by_authority(),
674                ),
675                submission_retriable_errors: aggregate_request_errors(
676                    retriable_errors_aggregator.status_by_authority(),
677                ),
678                observed_effects_digests: AggregatedEffectsDigests {
679                    digests: observed_effects_digests,
680                },
681            })
682        } else {
683            if observed_effects_digests.len() <= 1 {
684                debug_fatal!(
685                    "Expect at least 2 effects digests, but got {:?}",
686                    observed_effects_digests
687                );
688            }
689            Err(TransactionDriverError::ForkedExecution {
690                observed_effects_digests: AggregatedEffectsDigests {
691                    digests: observed_effects_digests,
692                },
693                submission_non_retriable_errors: aggregate_request_errors(
694                    non_retriable_errors_aggregator.status_by_authority(),
695                ),
696                submission_retriable_errors: aggregate_request_errors(
697                    retriable_errors_aggregator.status_by_authority(),
698                ),
699            })
700        }
701    }

703    #[instrument(level = "debug", skip_all, err(level = "debug"), ret, fields(validator_display_name = ?display_name))]
704    async fn wait_for_acknowledgment_rpc<A>(
705        &self,
706        name: AuthorityName,
707        display_name: String,
708        client: &Arc<SafeClient<A>>,
709        client_monitor: &Arc<ValidatorClientMonitor<A>>,
710        raw_request: &RawWaitForEffectsRequest,
711        options: &SubmitTransactionOptions,
712    ) -> Result<WaitForEffectsResponse, SuiError>
713    where
714        A: AuthorityAPI + Send + Sync + 'static + Clone,
715    {
716        let effects_start = Instant::now();
717        let backoff =
718            ExponentialBackoff::new(Duration::from_millis(100), MAX_WAIT_FOR_EFFECTS_RETRY_DELAY);
719        let ping_type = raw_request.get_ping_type();
720        // This loop should only retry errors that are retriable without new submission.
721        for (attempt, delay) in backoff.enumerate() {
722            let request: WaitForEffectsRequest = raw_request.clone().try_into().unwrap();
723            let result = client
724                .wait_for_effects(request, options.forwarded_client_addr)
725                .await;
726            match result {
727                Ok(response) => {
728                    let latency = effects_start.elapsed();
729                    client_monitor.record_interaction_result(OperationFeedback {
730                        authority_name: name,
731                        display_name: display_name.clone(),
732                        operation: OperationType::Effects,
733                        ping_type,
734                        result: Ok(latency),
735                    });
736                    return Ok(response);
737                }
738                Err(e) => {
739                    client_monitor.record_interaction_result(OperationFeedback {
740                        authority_name: name,
741                        display_name: display_name.clone(),
742                        operation: OperationType::Effects,
743                        ping_type,
744                        result: Err(()),
745                    });
746                    if !matches!(e.as_inner(), SuiErrorKind::RpcError(_, _)) {
747                        return Err(e);
748                    }
749                    tracing::trace!(
750                        ?name,
751                        "Wait for effects acknowledgement (attempt {attempt}): rpc error: {:?}",
752                        e
753                    );
754                }
755            };
756            sleep(delay).await;
757        }
758        Err(SuiErrorKind::TimeoutError.into())
759    }

761    /// Creates the final full response.
762    fn get_quorum_transaction_response(
763        &self,
764        effects_digest: TransactionEffectsDigest,
765        executed_data: Box<ExecutedData>,
766    ) -> QuorumTransactionResponse {
767        self.metrics.executed_transactions.inc();
768
769        tracing::debug!("Transaction executed with effects digest: {effects_digest}",);
770
771        let epoch = executed_data.effects.executed_epoch();
772        let details = FinalizedEffects {
773            effects: executed_data.effects,
774            finality_info: EffectsFinalityInfo::QuorumExecuted(epoch),
775        };
776
777        QuorumTransactionResponse {
778            effects: details,
779            events: executed_data.events,
780            input_objects: if !executed_data.input_objects.is_empty() {
781                Some(executed_data.input_objects)
782            } else {
783                None
784            },
785            output_objects: if !executed_data.output_objects.is_empty() {
786                Some(executed_data.output_objects)
787            } else {
788                None
789            },
790            auxiliary_data: None,
791        }
792    }
793}

795fn get_ping_type(tx_digest: &Option<TransactionDigest>, tx_type: TxType) -> Option<PingType> {
796    if tx_digest.is_none() {
797        Some(match tx_type {
798            TxType::SingleWriter => PingType::FastPath,
799            TxType::SharedObject => PingType::Consensus,
800        })
801    } else {
802        None
803    }
