4//! DNF bitmap index queries over ordered bucket streams.
5//!
6//! Callers build a `BitmapQuery` as an OR of terms. Each term is an AND of
7//! signed dimension-key literals. Evaluation yields matching bitmap members as
8//! they are produced. Back-pressure from downstream consumers, e.g. a
9//! `.take(page_size)`, propagates back to the backend-provided bucket streams
10//! and avoids materializing matches we won't use.
11//!
12//! Queries are intentionally restricted to anchored DNF: every term must contain
13//! at least one positive literal. Positive literals give the evaluator concrete
14//! bitmap streams to scan and intersect; negative literals only shrink those
15//! candidate streams. Supporting negative-only terms such as `NOT sender = A`
16//! would require scanning an external universe for the requested range and
17//! subtracting from it, which defeats the index's selective streaming behavior
18//! and makes pagination depend on a full-range scan. Requiring DNF at the API
19//! boundary also keeps this evaluator as a set of ordered stream merge-joins
20//! instead of a recursive expression engine or a query normalizer.
21//!
22//! Backends provide one ordered `(bucket_id, RoaringBitmap)` stream per
23//! dimension key. The merge-join machinery here is storage-agnostic: BigTable,
24//! RocksDB, or any other backend can reuse it as long as its bucket stream is
25//! sparse, ordered by the requested scan direction, and stores bitmap positions
26//! relative to that bucket.
27

42/// A stream of `(bucket_id, RoaringBitmap)` in the requested bucket order.
43/// Bitmap positions are **relative** to the bucket (u32 offsets `[0, BUCKET_SIZE)`)
44/// - edge trimming against the requested range happens at the flatten step.
45type BucketItem = Result<(u64, RoaringBitmap)>;
46pub type BucketStream = BoxStream<'static, BucketItem>;
47
48#[derive(Clone, Copy, Debug, Eq, PartialEq)]
49pub enum ScanDirection {
50    Ascending,
51    Descending,
52}
53
54impl ScanDirection {
55    pub fn is_ascending(self) -> bool {
56        matches!(self, Self::Ascending)
57    }
58}

60/// Storage backend that can scan one bitmap dimension key over a member range.
61///
62/// The returned stream must be sparse and ordered by the requested direction.
63/// Missing bucket rows are interpreted as all-zero bitmaps by the merge-join
64/// operators.
65pub trait BitmapBucketSource: Clone + Send + 'static {
66    fn scan_bucket_stream(
67        &self,
68        dimension_key: Vec<u8>,
69        range: Range<u64>,
70        direction: ScanDirection,
71    ) -> BucketStream;
72}

74/// A DNF query over bitmap dimension scans.
75///
76/// A query is a disjunction of terms. It must contain at least one term, and
77/// every term must be anchored by at least one included dimension key.
78#[derive(Clone, Debug)]
79pub struct BitmapQuery {
80    terms: Vec<BitmapTerm>,
81}
82
83/// One conjunction in a DNF bitmap query.
84///
85/// A term is a conjunction of signed literals. It must include at least one
86/// positive literal so the evaluator has a finite candidate stream to refine.
87#[derive(Clone, Debug)]
88pub struct BitmapTerm {
89    literals: Vec<BitmapLiteral>,
90}

92/// Validated `[dimension_tag][dimension_value]` lookup key.
93#[derive(Clone, Debug, Eq, PartialEq, Hash)]
94pub struct BitmapKey(Vec<u8>);
95
96/// One signed dimension-key literal in a bitmap term.
97#[derive(Clone, Debug)]
98pub enum BitmapLiteral {
99    Include(BitmapKey),
100    Exclude(BitmapKey),
101}
102
103impl BitmapKey {
104    pub fn new(bytes: Vec<u8>) -> Result<Self> {
105        if bytes.is_empty() {
106            bail!("bitmap dimension key must not be empty");
107        }
108        if bytes.len() == 1 {
109            bail!("bitmap dimension value must not be empty");
110        }
111        if IndexDimension::from_tag_byte(bytes[0]).is_none() {
112            bail!("unknown bitmap dimension tag {}", bytes[0]);
113        }
114        Ok(Self(bytes))
115    }
116
117    pub fn into_inner(self) -> Vec<u8> {
118        self.0
119    }
120
121    pub fn as_bytes(&self) -> &[u8] {
122        &self.0
123    }
124}
125
126impl TryFrom<Vec<u8>> for BitmapKey {
127    type Error = anyhow::Error;
128
129    fn try_from(value: Vec<u8>) -> Result<Self> {
130        Self::new(value)
131    }
132}
133
134impl BitmapLiteral {
135    pub fn include(dimension_key: Vec<u8>) -> Result<Self> {
136        Ok(Self::Include(BitmapKey::new(dimension_key)?))
137    }
138
139    pub fn exclude(dimension_key: Vec<u8>) -> Result<Self> {
140        Ok(Self::Exclude(BitmapKey::new(dimension_key)?))
141    }
142}

144impl BitmapQuery {
145    pub fn new(terms: Vec<BitmapTerm>) -> Result<Self> {
146        if terms.is_empty() {
147            bail!("bitmap query must contain at least one term");
148        }
149        Ok(Self { terms })
150    }
151
152    pub fn scan(dimension_key: Vec<u8>) -> Result<Self> {
153        Ok(Self {
154            terms: vec![BitmapTerm::new(vec![BitmapLiteral::include(
155                dimension_key,
156            )?])?],
157        })
158    }
159}

161impl BitmapTerm {
162    pub fn new(literals: Vec<BitmapLiteral>) -> Result<Self> {
163        if !literals
164            .iter()
165            .any(|literal| matches!(literal, BitmapLiteral::Include(_)))
166        {
167            bail!("bitmap query term must contain at least one include literal");
168        }
169        Ok(Self { literals })
170    }
171}

173/// Evaluate a DNF `BitmapQuery` against a backend-provided bitmap source.
174///
175/// This first evaluates the query into ordered bucket bitmaps, then flattens
176/// each relative bucket bitmap into absolute member ids inside `range`.
177pub fn eval_bitmap_query_stream<S>(
178    source: S,
179    query: BitmapQuery,
180    range: Range<u64>,
181    bucket_size: u64,
182    direction: ScanDirection,
183) -> BoxStream<'static, Result<u64>>
184where
185    S: BitmapBucketSource,
186{
187    let stream = eval_bitmap_query_bucket_stream(source, query, range.clone(), direction);
188    flatten_bucket_stream(stream, range, bucket_size, direction).boxed()
189}

191/// Evaluate a DNF `BitmapQuery` as an ordered stream of relative bucket bitmaps.
192pub fn eval_bitmap_query_bucket_stream<S>(
193    source: S,
194    query: BitmapQuery,
195    range: Range<u64>,
196    direction: ScanDirection,
197) -> BucketStream
198where
199    S: BitmapBucketSource,
200{
201    let streams: Vec<BucketStream> = query
202        .terms
203        .into_iter()
204        .map(|term| term_bucket_stream(source.clone(), term, range.clone(), direction).boxed())
205        .collect();
206    union_n(streams, direction).boxed()
207}

209/// Evaluate one DNF term: intersect all includes, then subtract excludes.
210fn term_bucket_stream<S>(
211    source: S,
212    term: BitmapTerm,
213    range: Range<u64>,
214    direction: ScanDirection,
215) -> impl Stream<Item = BucketItem> + Send + 'static
216where
217    S: BitmapBucketSource,
218{
219    let mut include = Vec::new();
220    let mut exclude = Vec::new();
221    for literal in term.literals {
222        match literal {
223            BitmapLiteral::Include(key) => include.push(key.into_inner()),
224            BitmapLiteral::Exclude(key) => exclude.push(key.into_inner()),
225        }
226    }
227
228    let include_streams: Vec<BucketStream> = include
229        .into_iter()
230        .map(|key| source.scan_bucket_stream(key, range.clone(), direction))
231        .collect();
232
233    let include_stream = intersect_n(include_streams, direction);
234
235    let exclude_streams: Vec<BucketStream> = exclude
236        .into_iter()
237        .map(|key| source.scan_bucket_stream(key, range.clone(), direction))
238        .collect();
239    let exclude_stream = union_n(exclude_streams, direction);
240
241    // Stream construction above is lazy. `subtract_two` polls both sides with
242    // `try_join!`, so the include intersection and exclude union are opened/read
243    // concurrently when this term stream is consumed.
244    subtract_two(include_stream, exclude_stream, direction)
245}

247/// Multi-way merge intersection over ordered bucket streams. Emits only those
248/// bucket_ids present in every child stream, with the bitwise AND of all their
249/// bitmaps. Drops empty results.
250pub fn intersect_n<S>(
251    streams: Vec<S>,
252    direction: ScanDirection,
253) -> impl Stream<Item = BucketItem> + Send + 'static
254where
255    S: Stream<Item = BucketItem> + Send + Unpin + 'static,
256{
257    async_stream::try_stream! {
258        if streams.is_empty() {
259            return;
260        }
261        let mut children: Vec<Peekable<S>> =
262            streams.into_iter().map(|s| s.peekable()).collect();
263
264        loop {
265            // Poll all child streams together so independent backend scans are
266            // opened/read concurrently instead of serializing one dimension at a
267            // time.
268            let Some((peeks, max_bucket)) = complete_peeks(peek_buckets(&mut children).await?)
269            else {
270                break;
271            };
272            let target_bucket = match direction {
273                ScanDirection::Ascending => max_bucket,
274                ScanDirection::Descending => peeks.iter().copied().min().expect("non-empty peeks"),
275            };
276
277            if peeks.iter().all(|&b| b == target_bucket) {
278                // All children at the same bucket: intersect and emit.
279                let mut acc: Option<RoaringBitmap> = None;
280                for child in children.iter_mut() {
281                    let (bid, bitmap) = Pin::new(child)
282                        .next()
283                        .await
284                        .expect("peek reported a value")?;
285                    debug_assert_eq!(bid, target_bucket);
286                    acc = Some(match acc {
287                        None => bitmap,
288                        Some(a) => a & bitmap,
289                    });
290                }
291                let bitmap = acc.expect("children non-empty");
292                if !bitmap.is_empty() {
293                    yield (target_bucket, bitmap);
294                }
295            } else {
296                // Sparse bitmap rows encode only non-empty buckets. If a child
297                // is behind the current alignment target, then at least one
298                // other include stream is already past that bucket and has an
299                // implicit all-zero bitmap there. The intersection for the
300                // lagging bucket is empty, so consume it and re-peek.
301                for (i, child) in children.iter_mut().enumerate() {
302                    let drop_bucket = match direction {
303                        ScanDirection::Ascending => peeks[i] < target_bucket,
304                        ScanDirection::Descending => peeks[i] > target_bucket,
305                    };
306                    if drop_bucket {
307                        let _ = Pin::new(child)
308                            .next()
309                            .await
310                            .expect("peek reported a value")?;
311                    }
312                }
313            }
314        }
315    }
316}

318/// Multi-way merge union over ordered bucket streams. Emits every bucket_id
319/// produced by any child, with the bitwise OR of the bitmaps at that bucket.
320pub fn union_n<S>(
321    streams: Vec<S>,
322    direction: ScanDirection,
323) -> impl Stream<Item = BucketItem> + Send + 'static
324where
325    S: Stream<Item = BucketItem> + Send + Unpin + 'static,
326{
327    async_stream::try_stream! {
328        if streams.is_empty() {
329            return;
330        }
331        let mut children: Vec<Peekable<S>> =
332            streams.into_iter().map(|s| s.peekable()).collect();
333
334        loop {
335            let peeks = peek_buckets(&mut children).await?;
336
337            // Evict exhausted children so their underlying streams (and any
338            // resources they own, such as request-scoped semaphore permits) are
339            // released promptly rather than held until the slowest peer finishes.
340            let mut surviving_children = Vec::with_capacity(children.len());
341            let mut surviving_peeks = Vec::with_capacity(peeks.len());
342            for (child, peek) in children.into_iter().zip_eq(peeks) {
343                if let Some(b) = peek {
344                    surviving_children.push(child);
345                    surviving_peeks.push(b);
346                }
347            }
348            children = surviving_children;
349
350            if surviving_peeks.is_empty() {
351                return;
352            }
353            let next_bucket = match direction {
354                ScanDirection::Ascending => *surviving_peeks
355                    .iter()
356                    .min()
357                    .expect("non-empty after evicting None peeks"),
358                ScanDirection::Descending => *surviving_peeks
359                    .iter()
360                    .max()
361                    .expect("non-empty after evicting None peeks"),
362            };
363
364            let mut acc: Option<RoaringBitmap> = None;
365            for (i, child) in children.iter_mut().enumerate() {
366                if surviving_peeks[i] == next_bucket {
367                    let (_, bitmap) = Pin::new(child)
368                        .next()
369                        .await
370                        .expect("peek reported a value")?;
371                    acc = Some(match acc {
372                        None => bitmap,
373                        Some(a) => a | bitmap,
374                    });
375                }
376            }
377            if let Some(bitmap) = acc
378                && !bitmap.is_empty()
379            {
380                yield (next_bucket, bitmap);
381            }
382        }
383    }
384}

386/// Merge-join subtraction for an anchored negative literal.
387///
388/// A negative literal in a valid term is evaluated as `a AND NOT b`, where `a`
389/// is the already-anchored positive candidate stream. Bitmap subtraction
390/// (`a_bm - b_bm`) has the same truth table without materializing `NOT b` over
391/// the whole range:
392///
393/// ```text
394/// a b | a AND NOT b
395/// 1 0 | 1
396/// 0 1 | 0
397/// 0 0 | 0
398/// 1 1 | 0
399/// ```
400///
401/// For each bucket in `a`, emits `a_bm - b_bm` if `b` has the same bucket,
402/// else emits `a_bm` unchanged. Drops empty results.
403pub fn subtract_two<A, B>(
404    a: A,
405    b: B,
406    direction: ScanDirection,
407) -> impl Stream<Item = BucketItem> + Send + 'static
408where
409    A: Stream<Item = BucketItem> + Send + 'static,
410    B: Stream<Item = BucketItem> + Send + 'static,
411{
412    async_stream::try_stream! {
413        let a = a.peekable();
414        let b = b.peekable();
415        futures::pin_mut!(a);
416        futures::pin_mut!(b);
417
418        loop {
419            // The peeks poll both streams concurrently and buffer their head
420            // rows; the later `next()` calls consume those buffered rows.
421            let (a_peek, b_peek) =
422                futures::try_join!(peek_bucket(a.as_mut()), peek_bucket(b.as_mut()))?;
423            let Some(a_bucket) = a_peek else {
424                return;
425            };
426
427            match b_peek {
428                None => {
429                    // No more negatives: flush a.
430                    let (bid, bitmap) = a
431                        .as_mut()
432                        .next()
433                        .await
434                        .expect("peek reported a value")?;
435                    if !bitmap.is_empty() {
436                        yield (bid, bitmap);
437                    }
438                }
439                Some(bb)
440                    if (direction.is_ascending() && bb > a_bucket)
441                        || (!direction.is_ascending() && bb < a_bucket) =>
442                {
443                    // b is ahead, emit a unchanged.
444                    let (bid, bitmap) = a
445                        .as_mut()
446                        .next()
447                        .await
448                        .expect("peek reported a value")?;
449                    if !bitmap.is_empty() {
450                        yield (bid, bitmap);
451                    }
452                }
453                Some(bb)
454                    if (direction.is_ascending() && bb < a_bucket)
455                        || (!direction.is_ascending() && bb > a_bucket) =>
456                {
457                    // b is behind; skip it.
458                    let _ = b
459                        .as_mut()
460                        .next()
461                        .await
462                        .expect("peek reported a value")?;
463                }
464                Some(_) => {
465                    // Same bucket: subtract.
466                    let (bid, a_bm) = a
467                        .as_mut()
468                        .next()
469                        .await
470                        .expect("peek reported a value")?;
471                    let (_, b_bm) = b
472                        .as_mut()
473                        .next()
474                        .await
475                        .expect("peek reported a value")?;
476                    let diff = a_bm - b_bm;
477                    if !diff.is_empty() {
478                        yield (bid, diff);
479                    }
480                }
481            }
482        }
483    }
484}

486/// Convert a stream of `(bucket_id, relative RoaringBitmap)` into absolute
487/// member ids, applying edge-bucket trimming against `range`.
488pub fn flatten_bucket_stream<S>(
489    stream: S,
490    range: Range<u64>,
491    bucket_size: u64,
492    direction: ScanDirection,
493) -> impl Stream<Item = Result<u64>> + Send + 'static
494where
495    S: Stream<Item = BucketItem> + Send + 'static,
496{
497    async_stream::try_stream! {
498        if range.is_empty() {
499            return;
500        }
501        let start_bucket = range.start / bucket_size;
502        let end_bucket = (range.end - 1) / bucket_size;
503        futures::pin_mut!(stream);
504        while let Some(item) = stream.next().await {
505            let (bucket_id, bitmap) = item?;
506            let bucket_start = bucket_id * bucket_size;
507            let is_first = bucket_id == start_bucket;
508            let is_last = bucket_id == end_bucket;
509            let lo = if is_first {
510                (range.start - bucket_start) as u32
511            } else {
512                0
513            };
514            let hi = if is_last {
515                ((range.end - bucket_start).min(bucket_size)) as u32
516            } else {
517                bucket_size as u32
518            };
519
520            if direction.is_ascending() {
521                for bit in bitmap.iter() {
522                    if bit >= lo && bit < hi {
523                        yield bucket_start + bit as u64;
524                    }
525                }
526            } else {
527                for bit in bitmap.iter().rev() {
528                    if bit >= lo && bit < hi {
529                        yield bucket_start + bit as u64;
530                    }
531                }
532            }
533        }
534    }
535}

537/// Peek at the head of a `Peekable<BucketStream>`, returning its bucket_id.
538/// If the stream's next item is an error, consumes it to propagate via `?`.
539async fn peek_bucket<S>(mut s: Pin<&mut Peekable<S>>) -> Result<Option<u64>>
540where
541    S: Stream<Item = BucketItem>,
542{
543    let peeked = match s.as_mut().peek().await {
544        None => None,
545        Some(Ok((b, _))) => Some(Ok(*b)),
546        Some(Err(_)) => Some(Err(())),
547    };
548
549    match peeked {
550        None => Ok(None),
551        Some(Ok(bucket)) => Ok(Some(bucket)),
552        Some(Err(())) => match s.as_mut().next().await {
553            Some(Err(e)) => Err(e),
554            Some(Ok((bucket, _))) => Ok(Some(bucket)),
555            None => Ok(None),
556        },
557    }
558}
559
560async fn peek_buckets<S>(streams: &mut [Peekable<S>]) -> Result<Vec<Option<u64>>>
561where
562    S: Stream<Item = BucketItem> + Unpin,
563{
564    futures::future::try_join_all(
565        streams
566            .iter_mut()
567            .map(|stream| peek_bucket(Pin::new(stream))),
568    )
569    .await
570}
571
572fn complete_peeks(peeks: Vec<Option<u64>>) -> Option<(Vec<u64>, u64)> {
573    let mut buckets = Vec::with_capacity(peeks.len());
574    let mut max_bucket = None;
575
576    for peek in peeks {
577        let bucket = peek?;
578        max_bucket = Some(max_bucket.map_or(bucket, |max: u64| max.max(bucket)));
579        buckets.push(bucket);
580    }
581
582    Some((buckets, max_bucket?))
583}

815    #[tokio::test]
816    async fn union_n_drops_exhausted_child() {
817        use std::sync::atomic::AtomicBool;
818        use std::sync::atomic::Ordering;
819        use std::task::Context;
820        use std::task::Poll;
821
822        struct ObserveDrop<S> {
823            inner: S,
824            dropped: Arc<AtomicBool>,
825        }
826
827        impl<S: Stream<Item = BucketItem> + Unpin> Stream for ObserveDrop<S> {
828            type Item = BucketItem;
829            fn poll_next(
830                mut self: Pin<&mut Self>,
831                cx: &mut Context<'_>,
832            ) -> Poll<Option<Self::Item>> {
833                Pin::new(&mut self.inner).poll_next(cx)
834            }
835        }
836
837        impl<S> Drop for ObserveDrop<S> {
838            fn drop(&mut self) {
839                self.dropped.store(true, Ordering::SeqCst);
840            }
841        }
842
843        let dropped = Arc::new(AtomicBool::new(false));
844        let short: BucketStream = ObserveDrop {
845            inner: stream::iter(vec![Ok::<_, anyhow::Error>((0u64, make_bitmap(&[1])))]),
846            dropped: dropped.clone(),
847        }
848        .boxed();
849        let long = make_bucket_stream(vec![(0, &[2]), (1, &[3]), (2, &[4])]);
850
851        let merged = union_n(vec![short, long], ScanDirection::Ascending);
852        futures::pin_mut!(merged);
853
854        // Bucket 0 merges both children; the short stream is now exhausted
855        // underneath but eviction has not run yet — it happens on the next
856        // peek_buckets call.
857        let (b, _) = merged.try_next().await.unwrap().unwrap();
858        assert_eq!(b, 0);
859        assert!(!dropped.load(Ordering::SeqCst));
860
861        // The iteration that yields bucket 1 first peeks, sees the short
862        // stream's peek is None, evicts it (dropping the Peekable and its
863        // inner ObserveDrop), then yields. The long stream still has bucket 2
864        // pending — proving the drop happened mid-merge, not at completion.
865        let (b, _) = merged.try_next().await.unwrap().unwrap();
866        assert_eq!(b, 1);
867        assert!(dropped.load(Ordering::SeqCst));
868
869        let (b, _) = merged.try_next().await.unwrap().unwrap();
870        assert_eq!(b, 2);
871        assert!(merged.try_next().await.unwrap().is_none());
872    }
