41/// The possible relationship types for a transaction: sent or affected.
42#[derive(Enum, Copy, Clone, Eq, PartialEq)]
43pub(crate) enum AddressTransactionRelationship {
44    /// Transactions this address has sent.
45    Sent,
46    /// Transactions that this address was involved in, either as the sender, sponsor, or as the owner of some object that was created, modified or transferred.
47    Affected,
48}
49
50/// Interface implemented by GraphQL types representing entities that are identified by an address.
51///
52/// An address uniquely represents either the public key of an account, or an object's ID, but never both. It is not possible to determine which type an address represents up-front. If an object is wrapped, its contents will not be accessible via its address, but it will still be possible to access other objects it owns.
53#[allow(clippy::duplicated_attributes)]
54#[derive(Interface)]
55#[graphql(
56    name = "IAddressable",
57    field(name = "address", ty = "SuiAddress"),
58    field(
59        name = "balance",
60        arg(name = "coin_type", ty = "TypeInput"),
61        ty = "Option<Result<Balance, RpcError<balance::Error>>>",
62        desc = "Fetch the total balance for coins with marker type `coinType` (e.g. `0x2::sui::SUI`), owned by this address.\n\nIf the address does not own any coins of that type, a balance of zero is returned.",
63    ),
64    field(
65        name = "balances",
66        arg(name = "first", ty = "Option<u64>"),
67        arg(name = "after", ty = "Option<balance::Cursor>"),
68        arg(name = "last", ty = "Option<u64>"),
69        arg(name = "before", ty = "Option<balance::Cursor>"),
70        ty = "Option<Result<Connection<String, Balance>, RpcError<balance::Error>>>",
71        desc = "Total balance across coins owned by this address, grouped by coin type.",
72    ),
73    field(
74        name = "default_suins_name",
75        ty = "Option<Result<String, RpcError>>",
76        desc = "The domain explicitly configured as the default SuiNS name for this address."
77    ),
78    field(
79        name = "multi_get_balances",
80        arg(name = "keys", ty = "Vec<TypeInput>"),
81        ty = "Option<Result<Vec<Balance>, RpcError<balance::Error>>>",
82        desc = "Fetch the total balances keyed by coin types (e.g. `0x2::sui::SUI`) owned by this address.\n\nReturns `null` when no checkpoint is set in scope (e.g. execution scope). If the address does not own any coins of a given type, a balance of zero is returned for that type.",
83    ),
84    field(
85        name = "objects",
86        arg(name = "first", ty = "Option<u64>"),
87        arg(name = "after", ty = "Option<object::CLive>"),
88        arg(name = "last", ty = "Option<u64>"),
89        arg(name = "before", ty = "Option<object::CLive>"),
90        arg(name = "filter", ty = "Option<ObjectFilter>"),
91        ty = "Option<Result<Connection<String, MoveObject>, RpcError<object::Error>>>",
92        desc = "Objects owned by this address, optionally filtered by type."
93    )
94)]
95pub(crate) enum IAddressable {
96    Address(Address),
97    CoinMetadata(CoinMetadata),
98    DynamicField(DynamicField),
99    MoveObject(MoveObject),
100    MovePackage(MovePackage),
101    Object(Object),
102    Validator(Validator),
103}

118    /// The Address' identifier, a 32-byte number represented as a 64-character hex string, with a lead "0x".
119    pub(crate) async fn address(&self) -> Result<SuiAddress, RpcError> {
120        Ok(self.address.into())
121    }

123    /// Attempts to fetch the object at this address.
124    pub(crate) async fn as_object(
125        &self,
126        ctx: &Context<'_>,
127    ) -> Option<Result<Object, RpcError<object::Error>>> {
128        Object::by_key(
129            ctx,
130            self.scope.clone(),
131            ObjectKey {
132                address: self.address.into(),
133                version: None,
134                root_version: self.scope.root_version().map(Into::into),
135                at_checkpoint: None,
136            },
137        )
138        .await
139        .transpose()
140    }

142    /// Fetch the total balance for coins with marker type `coinType` (e.g. `0x2::sui::SUI`), owned by this address.
143    ///
144    /// Returns `None` when no checkpoint is set in scope (e.g. execution scope).
145    /// If the address does not own any coins of that type, a balance of zero is returned.
146    pub(crate) async fn balance(
147        &self,
148        ctx: &Context<'_>,
149        coin_type: TypeInput,
150    ) -> Option<Result<Balance, RpcError<balance::Error>>> {
151        Balance::fetch_one(ctx, &self.scope, self.address, coin_type.into())
152            .await
153            .transpose()
154    }

156    /// Total balance across coins owned by this address, grouped by coin type.

162        last: Option<u64>,

166            async {

168                let limits = pagination.limits("IAddressable", "balances");
169                let page = Page::from_params(limits, first, after, last, before)?;
170                Balance::paginate(ctx, self.scope.clone(), self.address, page).await
171            }
172            .await,
173        )
174    }

298    /// Objects owned by this address, optionally filtered by type.
299    pub(crate) async fn objects(
300        &self,
301        ctx: &Context<'_>,
302        first: Option<u64>,
303        after: Option<object::CLive>,
304        last: Option<u64>,
305        before: Option<object::CLive>,
306        #[graphql(validator(custom = "OFValidator::allows_empty()"))] filter: Option<ObjectFilter>,
307    ) -> Option<Result<Connection<String, MoveObject>, RpcError<object::Error>>> {
308        Some(
309            async {
310                let pagination: &PaginationConfig = ctx.data()?;
311                let limits = pagination.limits("IAddressable", "objects");
312                let page = Page::from_params(limits, first, after, last, before)?;
313
314                // Create a filter that constrains to ADDRESS kind and this owner
315                let Some(filter) = filter.unwrap_or_default().intersect(ObjectFilter {
316                    owner_kind: Some(OwnerKind::Address),
317                    owner: Some(self.address.into()),
318                    ..Default::default()
319                }) else {
320                    return Ok(Connection::new(false, false));
321                };
322
323                let objects = Object::paginate_live(ctx, self.scope.clone(), page, filter).await?;
324                let mut move_objects =
325                    Connection::new(objects.has_previous_page, objects.has_next_page);
326
327                for edge in objects.edges {
328                    let move_obj = MoveObject::from_super(edge.node);
329                    move_objects.edges.push(Edge::new(edge.cursor, move_obj));
330                }
331
332                Ok(move_objects)
333            }
334            .await,
335        )
336    }

338    /// Transactions associated with this address.
339    ///
340    /// Similar behavior to the `transactions` in Query but supporting the additional `AddressTransactionRelationship` filter, which defaults to `SENT`.
341    pub(crate) async fn transactions(
342        &self,
343        ctx: &Context<'_>,
344        first: Option<u64>,
345        after: Option<CTransaction>,
346        last: Option<u64>,
347        before: Option<CTransaction>,
348        relation: Option<AddressTransactionRelationship>,
349        #[graphql(validator(custom = "TFValidator"))] filter: Option<TransactionFilter>,
350    ) -> Option<Result<Connection<String, Transaction>, RpcError>> {
351        Some(
352            async {
353                let pagination: &PaginationConfig = ctx.data()?;
354                let limits = pagination.limits("Address", "transactions");
355                let page = Page::from_params(limits, first, after, last, before)?;
356
357                // Default relation to SENT if not provided
358                let relation = relation.unwrap_or(AddressTransactionRelationship::Sent);
359
360                // Create address-specific filter based on relationship
361                let address_filter = match relation {
362                    AddressTransactionRelationship::Sent => TransactionFilter {
363                        sent_address: Some(self.address.into()),
364                        ..Default::default()
365                    },
366                    AddressTransactionRelationship::Affected => TransactionFilter {
367                        affected_address: Some(self.address.into()),
368                        ..Default::default()
369                    },
370                };
371
372                // Intersect with user-provided filter
373                let Some(filter) = filter.unwrap_or_default().intersect(address_filter) else {
374                    return Ok(Connection::new(false, false));
375                };
376
377                Transaction::paginate(ctx, self.scope.clone(), page, filter).await
378            }
379            .await,
380        )
381    }
